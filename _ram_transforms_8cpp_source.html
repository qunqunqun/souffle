177 bool HoistConditionsTransformer::hoistConditions(RamProgram& program) {
178  bool changed = false;
179 
180  // helper for collecting conditions from filter operations
181  auto addCondition = [](std::unique_ptr<RamCondition> condition,
182  std::unique_ptr<RamCondition> c) -> std::unique_ptr<RamCondition> {
183  if (condition == nullptr) {
184  return c;
185  } else {
186  return std::make_unique<RamConjunction>(std::move(condition), std::move(c));
187  }
188  };
189 
190  // hoist conditions to the most outer scope if they
191  // don't depend on RamTupleOperations
192  visitDepthFirst(program, [&](const RamQuery& query) {
193  std::unique_ptr<RamCondition> newCondition;
194  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
195  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
196  if (auto* filter = dynamic_cast<RamFilter*>(node.get())) {
197  const RamCondition& condition = filter->getCondition();
198  // if filter condition is independent of any RamTupleOperation,
199  // delete the filter operation and collect condition
200  if (rla->getLevel(&condition) == -1) {
201  changed = true;
202  newCondition = addCondition(std::move(newCondition), souffle::clone(&condition));
203  node->apply(makeLambdaRamMapper(filterRewriter));
204  return souffle::clone(&filter->getOperation());
205  }
206  }
207  node->apply(makeLambdaRamMapper(filterRewriter));
208  return node;
209  };
210  auto* mQuery = const_cast<RamQuery*>(&query);
211  mQuery->apply(makeLambdaRamMapper(filterRewriter));
212  if (newCondition != nullptr) {
213  // insert new filter operation at outer-most level of the query
214  changed = true;
215  auto* nestedOp = const_cast<RamOperation*>(&mQuery->getOperation());
216  mQuery->rewrite(
217  nestedOp, std::make_unique<RamFilter>(std::move(newCondition), souffle::clone(nestedOp)));
218  }
219  });
220 
221  // hoist conditions for each RamTupleOperation operation
222  visitDepthFirst(program, [&](const RamTupleOperation& search) {
223  std::unique_ptr<RamCondition> newCondition;
224  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
225  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
226  if (auto* filter = dynamic_cast<RamFilter*>(node.get())) {
227  const RamCondition& condition = filter->getCondition();
228  // if filter condition matches level of RamTupleOperation,
229  // delete the filter operation and collect condition
230  if (rla->getLevel(&condition) == search.getTupleId()) {
231  changed = true;
232  newCondition = addCondition(std::move(newCondition), souffle::clone(&condition));
233  node->apply(makeLambdaRamMapper(filterRewriter));
234  return souffle::clone(&filter->getOperation());
235  }
236  }
237  node->apply(makeLambdaRamMapper(filterRewriter));
238  return node;
239  };
240  auto* tupleOp = const_cast<RamTupleOperation*>(&search);
241  tupleOp->apply(makeLambdaRamMapper(filterRewriter));
242  if (newCondition != nullptr) {
243  // insert new filter operation after the search operation
244  changed = true;
245  tupleOp->rewrite(&tupleOp->getOperation(), std::make_unique<RamFilter>(std::move(newCondition),
246  souffle::clone(&tupleOp->getOperation())));
247  }
248  });
249  return changed;
250 }

253 // Retrieves the <expr1> <= Tuple[level, element] <= <expr2> part of the constraint as a pair { <expr1>,
254 // <expr2> }
255 ExpressionPair MakeIndexTransformer::getLowerUpperExpression(
256  RamCondition* c, size_t& element, int identifier) {
257  if (auto* binRelOp = dynamic_cast<RamConstraint*>(c)) {
258  // TODO: FIXME: how does this interact w/ `FEQ`?
259 
260  if (isEqConstraint(binRelOp->getOperator())) {
261  if (const auto* lhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getLHS())) {
262  const RamExpression* rhs = &binRelOp->getRHS();
263  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
264  element = lhs->getElement();
265  return {clone(rhs), clone(rhs)};
266  }
267  }
268  if (const auto* rhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getRHS())) {
269  const RamExpression* lhs = &binRelOp->getLHS();
270  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
271  element = rhs->getElement();
272  return {clone(lhs), clone(lhs)};
273  }
274  }
275  }
276 
277  if (isLessEqualConstraint(binRelOp->getOperator())) {
278  // Tuple[level, element] <= <expr>
279  if (const auto* lhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getLHS())) {
280  const RamExpression* rhs = &binRelOp->getRHS();
281  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
282  element = lhs->getElement();
283  return {std::make_unique<RamUndefValue>(), clone(rhs)};
284  }
285  }
286  // <expr> <= Tuple[level, element]
287  if (const auto* rhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getRHS())) {
288  const RamExpression* lhs = &binRelOp->getLHS();
289  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
290  element = rhs->getElement();
291  return {clone(lhs), std::make_unique<RamUndefValue>()};
292  }
293  }
294  }
295 
296  if (isGreaterEqualConstraint(binRelOp->getOperator())) {
297  // Tuple[level, element] >= <expr>
298  if (const auto* lhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getLHS())) {
299  const RamExpression* rhs = &binRelOp->getRHS();
300  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
301  element = lhs->getElement();
302  return {clone(rhs), std::make_unique<RamUndefValue>()};
303  }
304  }
305  // <expr> >= Tuple[level, element]
306  if (const auto* rhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getRHS())) {
307  const RamExpression* lhs = &binRelOp->getLHS();
308  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
309  element = rhs->getElement();
310  return {std::make_unique<RamUndefValue>(), clone(lhs)};
311  }
312  }
313  }
314  if (isLessThanConstraint(binRelOp->getOperator())) {
315  // Tuple[level, element] < <expr>
316  // is equivalent to
317  // Tuple[level, element] <= <expr> - 1
318  if (const auto* lhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getLHS())) {
319  const RamExpression* rhs = &binRelOp->getRHS();
320  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
321  element = lhs->getElement();
322  std::vector<std::unique_ptr<RamExpression>> expressions;
323  expressions.push_back(souffle::clone(rhs));
324  expressions.push_back(std::make_unique<RamSignedConstant>(RamDomain(1)));
325 
326  return {std::make_unique<RamUndefValue>(),
327  std::make_unique<RamIntrinsicOperator>(FunctorOp::SUB, std::move(expressions))};
328  }
329  }
330  // <expr> < Tuple[level, element]
331  // is equivalent to
332  // <expr> + 1 <= Tuple[level, element]
333  if (const auto* rhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getRHS())) {
334  const RamExpression* lhs = &binRelOp->getLHS();
335  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
336  element = rhs->getElement();
337  std::vector<std::unique_ptr<RamExpression>> expressions;
338  expressions.push_back(souffle::clone(lhs));
339  expressions.push_back(std::make_unique<RamSignedConstant>(RamDomain(1)));
340 
341  return {std::make_unique<RamIntrinsicOperator>(FunctorOp::ADD, std::move(expressions)),
342  std::make_unique<RamUndefValue>()};
343  }
344  }
345  }
346 
347  if (isGreaterThanConstraint(binRelOp->getOperator())) {
348  // Tuple[level, element] > <expr>
349  // is equivalent to
350  // Tuple[level, element] >= <expr> + 1
351  if (const auto* lhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getLHS())) {
352  const RamExpression* rhs = &binRelOp->getRHS();
353  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
354  element = lhs->getElement();
355  std::vector<std::unique_ptr<RamExpression>> expressions;
356  expressions.push_back(souffle::clone(rhs));
357  expressions.push_back(std::make_unique<RamSignedConstant>(RamDomain(1)));
358 
359  return {std::make_unique<RamIntrinsicOperator>(FunctorOp::ADD, std::move(expressions)),
360  std::make_unique<RamUndefValue>()};
361  }
362  }
363  // <expr> > Tuple[level, element]
364  // is equivalent to
365  // <expr> - 1 >= Tuple[level, element]
366  if (const auto* rhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getRHS())) {
367  const RamExpression* lhs = &binRelOp->getLHS();
368  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
369  element = rhs->getElement();
370  std::vector<std::unique_ptr<RamExpression>> expressions;
371  expressions.push_back(souffle::clone(lhs));
372  expressions.push_back(std::make_unique<RamSignedConstant>(RamDomain(1)));
373 
374  return {std::make_unique<RamUndefValue>(),
375  std::make_unique<RamIntrinsicOperator>(FunctorOp::SUB, std::move(expressions))};
376  }
377  }
378  }
379  }
380  return {std::make_unique<RamUndefValue>(), std::make_unique<RamUndefValue>()};
381 }

383 std::unique_ptr<RamCondition> MakeIndexTransformer::constructPattern(RamPattern& queryPattern,
384  bool& indexable, std::vector<std::unique_ptr<RamCondition>> conditionList, int identifier) {
385  // Remaining conditions which cannot be handled by an index
386  std::unique_ptr<RamCondition> condition;
387  auto addCondition = [&](std::unique_ptr<RamCondition> c) {
388  if (condition != nullptr) {
389  condition = std::make_unique<RamConjunction>(std::move(condition), std::move(c));
390  } else {
391  condition = std::move(c);
392  }
393  };
394 
395  // Build query pattern and remaining condition
396  for (auto& cond : conditionList) {
397  size_t element = 0;
398  std::unique_ptr<RamExpression> lowerExpression;
399  std::unique_ptr<RamExpression> upperExpression;
400  std::tie(lowerExpression, upperExpression) = getLowerUpperExpression(cond.get(), element, identifier);
401 
402  // we have new bounds if both are not nullptr
403  if (!isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get())) {
404  // if no previous bounds are set then just assign them, consider both bounds to be set (but not
405  // necessarily defined) in all remaining cases
406  indexable = true;
407  if (isRamUndefValue(queryPattern.first[element].get()) &&
408  isRamUndefValue(queryPattern.second[element].get())) {
409  queryPattern.first[element] = std::move(lowerExpression);
410  queryPattern.second[element] = std::move(upperExpression);
411  // if lower bound is undefined and we have a new lower bound then assign it
412  } else if (isRamUndefValue(queryPattern.first[element].get()) &&
413  !isRamUndefValue(lowerExpression.get()) && isRamUndefValue(upperExpression.get())) {
414  queryPattern.first[element] = std::move(lowerExpression);
415  // if upper bound is undefined and we have a new upper bound then assign it
416  } else if (isRamUndefValue(queryPattern.second[element].get()) &&
417  isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get())) {
418  queryPattern.second[element] = std::move(upperExpression);
419  // if both bounds are defined ...
420  // and equal then we have a previous equality constraint i.e. Tuple[level, element] = <expr1>
421  } else if (!isRamUndefValue(queryPattern.first[element].get()) &&
422  !isRamUndefValue(queryPattern.second[element].get()) &&
423  (*(queryPattern.first[element]) == *(queryPattern.second[element]))) {
424  // new equality constraint i.e. Tuple[level, element] = <expr2>
425  // simply hoist <expr1> = <expr2> to the outer loop
426  if (!isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get())) {
427  // FIXME: `FEQ` handling; need to know if the expr is a float exp or not
428  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::EQ,
429  souffle::clone(queryPattern.first[element]), std::move(lowerExpression)));
430  }
431  // new lower bound i.e. Tuple[level, element] >= <expr2>
432  // we need to hoist <expr1> >= <expr2> to the outer loop
433  else if (!isRamUndefValue(lowerExpression.get()) && isRamUndefValue(upperExpression.get())) {
434  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::GE,
435  souffle::clone(queryPattern.first[element]), std::move(lowerExpression)));
436  }
437  // new upper bound i.e. Tuple[level, element] <= <expr2>
438  // we need to hoist <expr1> <= <expr2> to the outer loop
439  else if (isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get())) {
440  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::LE,
441  souffle::clone(queryPattern.first[element]), std::move(upperExpression)));
442  }
443  // if either bound is defined but they aren't equal we must consider the cases for updating
444  // them note that at this point we know that if we have a lower/upper bound it can't be the
445  // first one
446  } else if (!isRamUndefValue(queryPattern.first[element].get()) ||
447  !isRamUndefValue(queryPattern.second[element].get())) {
448  // if we have a new equality constraint and previous inequality constraints
449  if (!isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get()) &&
450  *lowerExpression == *upperExpression) {
451  // if Tuple[level, element] >= <expr1> and we see Tuple[level, element] = <expr2>
452  // need to hoist <expr2> >= <expr1> to the outer loop
453  if (!isRamUndefValue(queryPattern.first[element].get())) {
454  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::GE,
455  souffle::clone(lowerExpression), std::move(queryPattern.first[element])));
456  }
457  // if Tuple[level, element] <= <expr1> and we see Tuple[level, element] = <expr2>
458  // need to hoist <expr2> <= <expr1> to the outer loop
459  if (!isRamUndefValue(queryPattern.second[element].get())) {
460  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::LE,
461  souffle::clone(upperExpression), std::move(queryPattern.second[element])));
462  }
463  // finally replace bounds with equality constraint
464  queryPattern.first[element] = std::move(lowerExpression);
465  queryPattern.second[element] = std::move(upperExpression);
466  // if we have a new lower bound
467  } else if (!isRamUndefValue(lowerExpression.get())) {
468  // we want the tightest lower bound so we take the max
469  std::vector<std::unique_ptr<RamExpression>> maxArguments;
470  maxArguments.push_back(std::move(queryPattern.first[element]));
471  maxArguments.push_back(std::move(lowerExpression));
472 
473  queryPattern.first[element] =
474  std::make_unique<RamIntrinsicOperator>(FunctorOp::MAX, std::move(maxArguments));
475  // if we have a new upper bound
476  } else if (!isRamUndefValue(upperExpression.get())) {
477  // we want the tightest upper bound so we take the min
478  std::vector<std::unique_ptr<RamExpression>> minArguments;
479  minArguments.push_back(std::move(queryPattern.second[element]));
480  minArguments.push_back(std::move(upperExpression));
481 
482  queryPattern.second[element] =
483  std::make_unique<RamIntrinsicOperator>(FunctorOp::MIN, std::move(minArguments));
484  }
485  }
486  } else {
487  addCondition(std::move(cond));
488  }
489  }
490 
491  // Avoid null-pointers for condition and query pattern
492  if (condition == nullptr) {
493  condition = std::make_unique<RamTrue>();
494  }
495  return condition;
496 }

521 std::unique_ptr<RamOperation> MakeIndexTransformer::rewriteScan(const RamScan* scan) {
522  if (const auto* filter = dynamic_cast<const RamFilter*>(&scan->getOperation())) {
523  const RamRelation& rel = scan->getRelation();
524  const int identifier = scan->getTupleId();

606  bool changed = false;
607 
608  // helper for collecting conditions from filter operations
609  auto addCondition = [](std::unique_ptr<RamCondition> condition,
610  std::unique_ptr<RamCondition> c) -> std::unique_ptr<RamCondition> {
611  if (condition == nullptr) {
612  return c;
613  } else {
614  return std::make_unique<RamConjunction>(std::move(condition), std::move(c));
615  }
616  };

618  visitDepthFirst(program, [&](const RamQuery& query) {
619  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> indexToFilterRewriter =
620  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
621  // find a RamIndexOperation
622  if (const RamIndexOperation* indexOperation = dynamic_cast<RamIndexOperation*>(node.get())) {
623  auto attributesToDischarge =
624  idxAnalysis->getIndexes(indexOperation->getRelation()).getAttributesToDischarge();
625 
626  auto pattern = indexOperation->getRangePattern();
627  std::unique_ptr<RamCondition> condition;
628  RamPattern updatedPattern;
629

631  updatedPattern.first.emplace_back(p->clone());
632  }
633  for (RamExpression* p : indexOperation->getRangePattern().second) {
634  updatedPattern.second.emplace_back(p->clone());

639  std::unique_ptr<RamConstraint> lowerBound;
640  std::unique_ptr<RamConstraint> upperBound;
641  changed = true;

651  upperBound = std::make_unique<RamConstraint>(BinaryConstraintOp::LE,

653  souffle::clone(pattern.second[i]));

659  updatedPattern.second[i] = std::make_unique<RamUndefValue>();

676  pscan->getTupleId(), std::move(updatedPattern), std::move(filter),

683  } else if (const RamIndexAggregate* iagg = dynamic_cast<RamIndexAggregate*>(node.get())) {

989 bool HoistAggregateTransformer::hoistAggregate(RamProgram& program) {
990  bool changed = false;
991 
992  // There are two cases: aggregates that have no data-dependencies on
993  // other RAM operations, and aggregates that have data-dependencies.
994  // A rewriter has two tasks: (1) identify a single aggregate that
995  // can be hoisted and (2) insert it at the outermost level.
996  // We assume all RamOperations are renumbered for this transformation.
997 
998  // Hoist a single aggregate to an outer scope that is data-independent.
999  visitDepthFirst(program, [&](const RamQuery& query) {
1000  std::unique_ptr<RamNestedOperation> newAgg;
1001  bool priorTupleOp = false;
1002  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> aggRewriter =
1003  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
1004  if (nullptr != dynamic_cast<RamAggregate*>(node.get())) {
1005  auto* tupleOp = dynamic_cast<RamTupleOperation*>(node.get());
1006  assert(tupleOp != nullptr && "aggregate conversion to tuple operation failed");
1007  if (rla->getLevel(tupleOp) == -1 && !priorTupleOp) {
1008  changed = true;
1009  newAgg = souffle::clone(tupleOp);
1010  assert(newAgg != nullptr && "failed to make a clone");
1011  return souffle::clone(&tupleOp->getOperation());
1012  }
1013  } else if (nullptr != dynamic_cast<RamTupleOperation*>(node.get())) {
1014  // tuple operation that is a non-aggregate
1015  priorTupleOp = true;
1016  }
1017  node->apply(makeLambdaRamMapper(aggRewriter));
1018  return node;
1019  };
1020  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(aggRewriter));
1021  if (newAgg != nullptr) {
1022  newAgg->rewrite(&newAgg->getOperation(), souffle::clone(&query.getOperation()));
1023  const_cast<RamQuery*>(&query)->rewrite(&query.getOperation(), std::move(newAgg));
1024  }
1025  });
1026 
1027  // hoist a single aggregate to an outer scope that is data-dependent on a prior operation.
1028  visitDepthFirst(program, [&](const RamQuery& query) {
1029  int newLevel = -1;
1030  std::unique_ptr<RamNestedOperation> newAgg;
1031  int priorOpLevel = -1;
1032 
1033  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> aggRewriter =
1034  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
1035  if (nullptr != dynamic_cast<RamAbstractAggregate*>(node.get())) {
1036  auto* tupleOp = dynamic_cast<RamTupleOperation*>(node.get());
1037  assert(tupleOp != nullptr && "aggregate conversion to nested operation failed");
1038  int dataDepLevel = rla->getLevel(tupleOp);
1039  if (dataDepLevel != -1 && dataDepLevel < tupleOp->getTupleId() - 1) {
1040  // If all tuple ops between the data-dependence level and agg
1041  // are aggregates, then we do not hoist, i.e., we would
1042  // continuously swap their positions.
1043  if (dataDepLevel != priorOpLevel) {
1044  changed = true;
1045  newLevel = dataDepLevel;
1046  newAgg = souffle::clone(tupleOp);
1047  assert(newAgg != nullptr && "failed to make a clone");
1048  return souffle::clone(&tupleOp->getOperation());
1049  }
1050  }
1051  } else if (const RamTupleOperation* tupleOp = dynamic_cast<RamTupleOperation*>(node.get())) {
1052  priorOpLevel = tupleOp->getTupleId();
1053  }
1054  node->apply(makeLambdaRamMapper(aggRewriter));
1055  if (auto* search = dynamic_cast<RamTupleOperation*>(node.get())) {
1056  if (newAgg != nullptr && search->getTupleId() == newLevel) {
1057  newAgg->rewrite(&newAgg->getOperation(), souffle::clone(&search->getOperation()));
1058  search->rewrite(&search->getOperation(), std::move(newAgg));
1059  }
1060  }
1061  return node;
1062  };
1063  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(aggRewriter));
1064  });
1065  return changed;
1066 } // namespace souffle
