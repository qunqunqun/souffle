53  bool setAttributeNames(AstTranslationUnit& translationUnit) {
54  bool changed = false;
55  AstProgram* program = translationUnit.getProgram();
56  for (AstIO* io : program->getIOs()) {
57  if (io->hasDirective("attributeNames")) {
58  continue;
59  }
60  AstRelation* rel = getRelation(*translationUnit.getProgram(), io->getQualifiedName());
61  std::string delimiter("\t");
62  if (io->hasDirective("delimiter")) {
63  delimiter = io->getDirective("delimiter");
64  }
65 
66  std::vector<std::string> attributeNames;
67  for (const auto* attribute : rel->getAttributes()) {
68  attributeNames.push_back(attribute->getName());
69  }
70 
71  if (Global::config().has("provenance")) {
72  auto auxArityAnalysis = translationUnit.getAnalysis<AuxiliaryArity>();
73  std::vector<std::string> originalAttributeNames(
74  attributeNames.begin(), attributeNames.end() - auxArityAnalysis->getArity(rel));
75  io->addDirective("attributeNames", toString(join(originalAttributeNames, delimiter)));
76  } else {
77  io->addDirective("attributeNames", toString(join(attributeNames, delimiter)));
78  }
79  changed = true;
80  }
81  return changed;
82  }

84  bool setAttributeTypes(AstTranslationUnit& translationUnit) {
85  bool changed = false;
86  AstProgram* program = translationUnit.getProgram();
87  auto auxArityAnalysis = translationUnit.getAnalysis<AuxiliaryArity>();
88  auto typeEnv = &translationUnit.getAnalysis<TypeEnvironmentAnalysis>()->getTypeEnvironment();
89 
90  for (AstIO* io : program->getIOs()) {
91  AstRelation* rel = getRelation(*translationUnit.getProgram(), io->getQualifiedName());
92  // Prepare type system information.
93  std::vector<std::string> attributesTypes;
94 
95  for (const auto* attribute : rel->getAttributes()) {
96  auto typeName = attribute->getTypeName();
97  auto type = getTypeQualifier(typeEnv->getType(typeName));
98  attributesTypes.push_back(type);
99  }
100 
101  // Casting due to json11.h type requirements.
102  long long arity{static_cast<long long>(rel->getArity() - auxArityAnalysis->getArity(rel))};
103  long long auxArity{static_cast<long long>(auxArityAnalysis->getArity(rel))};
104 
105  Json relJson = Json::object{{"arity", arity}, {"auxArity", auxArity},
106  {"types", Json::array(attributesTypes.begin(), attributesTypes.end())}};
107 
108  Json types = Json::object{{"relation", relJson}, {"records", getRecordsTypes(translationUnit)}};
109 
110  io->addDirective("types", types.dump());
111  changed = true;
112  }
113  return changed;
114  }

120  Json getRecordsTypes(AstTranslationUnit& translationUnit) const {
121  static Json ramRecordTypes;
122  // Check if the types where already constructed
123  if (!ramRecordTypes.is_null()) {
124  return ramRecordTypes;
125  }
126 
127  AstProgram* program = translationUnit.getProgram();
128  auto typeEnv = &translationUnit.getAnalysis<TypeEnvironmentAnalysis>()->getTypeEnvironment();
129  std::vector<std::string> elementTypes;
130  std::map<std::string, Json> records;
131 
132  // Iterate over all record types in the program populating the records map.
133  for (auto* astType : program->getTypes()) {
134  const auto& type = typeEnv->getType(astType->getQualifiedName());
135  if (isA<RecordType>(type)) {
136  elementTypes.clear();
137 
138  for (const Type* field : as<RecordType>(type)->getFields()) {
139  elementTypes.push_back(getTypeQualifier(*field));
140  }
141  const size_t recordArity = elementTypes.size();
142  Json recordInfo = Json::object{
143  {"types", std::move(elementTypes)}, {"arity", static_cast<long long>(recordArity)}};
144  records.emplace(getTypeQualifier(type), std::move(recordInfo));
145  }
146  }
147 
148  ramRecordTypes = Json(records);
149  return ramRecordTypes;
150  }
