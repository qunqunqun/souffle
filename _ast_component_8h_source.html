39 /**
40  * A component type is
41  *
42  * name < Type1, Type2, ... >
43  *
44  * where name is the component name and < Type, Type, ... > is a
45  * list of component type parameters (either actual or formal).
46  */
47 class AstComponentType : public AstNode {
48 public:
49  AstComponentType(std::string name = "", std::vector<AstQualifiedName> params = {}, SrcLocation loc = {})
50  : AstNode(std::move(loc)), name(std::move(name)), typeParams(std::move(params)) {}
51 
52  /** get component name */
53  const std::string& getName() const {
54  return name;
55  }
56 
57  /** set component name */
58  void setName(std::string n) {
59  name = std::move(n);
60  }
61 
62  /** get component type parameters */
63  const std::vector<AstQualifiedName>& getTypeParameters() const {
64  return typeParams;
65  }
66 
67  /** set component type parameters */
68  void setTypeParameters(const std::vector<AstQualifiedName>& params) {
69  typeParams = params;
70  }
71 
72  AstComponentType* clone() const override {
73  return new AstComponentType(name, typeParams, getSrcLoc());
74  }
75 
76 protected:
77  void print(std::ostream& os) const override {
78  os << name;
79  if (!typeParams.empty()) {
80  os << "<" << join(typeParams) << ">";
81  }
82  }
83 
84  bool equal(const AstNode& node) const override {
85  const auto& other = static_cast<const AstComponentType&>(node);
86  return name == other.name && typeParams == other.typeParams;
87  }
88 
89 private:
90  /** component name */
91  std::string name;
92 
93  /** component type parameters */
94  std::vector<AstQualifiedName> typeParams;
95 };
