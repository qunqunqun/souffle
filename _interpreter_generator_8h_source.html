62  /**
63  * @brief Generate the tree based on given entry.
64  * Return a NodePtr to the root.
65  */
66  NodePtr generateTree(const RamNode& root, const RamProgram& program) {
67  this->program = const_cast<RamProgram*>(&program);
68  // Encode all relation, indexPos and viewId.
69  visitDepthFirst(root, [&](const RamNode& node) {
70  if (dynamic_cast<const RamQuery*>(&node) != nullptr) {
71  newQueryBlock();
72  }
73  if (const auto* indexSearch = dynamic_cast<const RamIndexOperation*>(&node)) {
74  encodeIndexPos(*indexSearch);
75  encodeView(indexSearch);
76  } else if (const auto* exists = dynamic_cast<const RamExistenceCheck*>(&node)) {
77  encodeIndexPos(*exists);
78  encodeView(exists);
79  } else if (const auto* provExists = dynamic_cast<const RamProvenanceExistenceCheck*>(&node)) {
80  encodeIndexPos(*provExists);
81  encodeView(provExists);
82  }
83  });
84  // Parse program
85  return visit(root);
86  }

504  NodePtr visitQuery(const RamQuery& query) override {
505  std::shared_ptr<InterpreterPreamble> preamble = std::make_shared<InterpreterPreamble>();
506  parentQueryPreamble = preamble;
507  // split terms of conditions of outer-most filter operation
508  // into terms that require a context and terms that
509  // do not require a view
510  const RamOperation* next = &query.getOperation();
511  std::vector<const RamCondition*> freeOfView;
512  if (const auto* filter = dynamic_cast<const RamFilter*>(&query.getOperation())) {
513  next = &filter->getOperation();
514  // Check terms of outer filter operation whether they can be pushed before
515  // the view-generation for speed improvements
516  auto conditions = toConjunctionList(&filter->getCondition());
517  for (auto const& cur : conditions) {
518  bool needView = false;
519  visitDepthFirst(*cur, [&](const RamNode& node) {
520  if (requireView(&node)) {
521  needView = true;
522  const auto& rel = getRelationRefForView(&node);
523  preamble->addViewInfoForFilter(
524  encodeRelation(rel), indexTable[&node], encodeView(&node));
525  }
526  });
527 
528  if (needView) {
529  preamble->addViewOperationForFilter(visit(*cur));
530  } else {
531  preamble->addViewFreeOperationForFilter(visit(*cur));
532  }
533  }
534  }
535 
536  visitDepthFirst(*next, [&](const RamNode& node) {
537  if (requireView(&node)) {
538  const auto& rel = getRelationRefForView(&node);
539  preamble->addViewInfoForNested(encodeRelation(rel), indexTable[&node], encodeView(&node));
540  };
541  });
542 
543  visitDepthFirst(*next, [&](const RamAbstractParallel&) { preamble->isParallel = true; });
544 
545  NodePtrVec children;
546  children.push_back(visit(*next));
547 
548  auto res = std::make_unique<InterpreterNode>(I_Query, &query, std::move(children));
549  res->setPreamble(parentQueryPreamble);
550  return res;
551  }

625  /** @brief Return operation index id from the result of indexAnalysis */
626  template <class RamNode>
627  size_t encodeIndexPos(RamNode& node) {
628  const MinIndexSelection& orderSet = isa->getIndexes(node.getRelation());
629  SearchSignature signature = isa->getSearchSignature(&node);
630  // A zero signature is equivalent as a full order signature.
631  if (signature.empty()) {

642  if (pos != viewTable.end()) {
643  return pos->second;

702  /**
703  * @brief Convert terms of a conjunction to a list
704  *
705  * Convert a condition of the format C1 /\ C2 /\ ... /\ Cn
706  * to a list {C1, C2, ..., Cn}.
707  */
708  inline std::vector<const RamCondition*> toConjunctionList(const RamCondition* condition) {
709  std::vector<const RamCondition*> conditionList;
710  std::queue<const RamCondition*> conditionsToProcess;
711  if (condition != nullptr) {
712  conditionsToProcess.push(condition);
713  while (!conditionsToProcess.empty()) {
714  condition = conditionsToProcess.front();
715  conditionsToProcess.pop();
716  if (const auto* ramConj = dynamic_cast<const RamConjunction*>(condition)) {
717  conditionsToProcess.push(&ramConj->getLHS());
718  conditionsToProcess.push(&ramConj->getRHS());
719  } else {
720  conditionList.emplace_back(condition);
721  }
722  }
723  }
724  return conditionList;
725  }

727  void createRelation(const RamRelation& id, const MinIndexSelection& orderSet, const size_t idx) {
728  RelationHandle res;
729  if (relations.size() < idx + 1) {
730  relations.resize(idx + 1);
731  }
732  if (id.getRepresentation() == RelationRepresentation::EQREL) {
733  res = std::make_unique<InterpreterEqRelation>(id.getArity(), id.getAuxiliaryArity(), id.getName(),
734  std::vector<std::string>(), orderSet);
735  } else {
736  if (isProvenance) {
737  res = std::make_unique<InterpreterRelation>(id.getArity(), id.getAuxiliaryArity(),
738  id.getName(), std::vector<std::string>(), orderSet, createBTreeProvenanceIndex);
739  } else {
740  res = std::make_unique<InterpreterRelation>(id.getArity(), id.getAuxiliaryArity(),
741  id.getName(), std::vector<std::string>(), orderSet);
742  }
743  }
744  relations[idx] = std::make_unique<RelationHandle>(std::move(res));
745  }
