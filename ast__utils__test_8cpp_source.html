47 TEST(AstUtils, Grounded) {
48  // create an example clause:
49  auto clause = std::make_unique<AstClause>();
50 
51  // something like:
52  // r(X,Y,Z) :- a(X), X = Y, !b(Z).
53 
54  // r(X,Y,Z)
55  auto* head = new AstAtom("r");
56  head->addArgument(std::unique_ptr<AstArgument>(new AstVariable("X")));
57  head->addArgument(std::unique_ptr<AstArgument>(new AstVariable("Y")));
58  head->addArgument(std::unique_ptr<AstArgument>(new AstVariable("Z")));
59  clause->setHead(std::unique_ptr<AstAtom>(head));
60 
61  // a(X)
62  auto* a = new AstAtom("a");
63  a->addArgument(std::unique_ptr<AstArgument>(new AstVariable("X")));
64  clause->addToBody(std::unique_ptr<AstLiteral>(a));
65 
66  // X = Y
67  AstLiteral* e1 = new AstBinaryConstraint(BinaryConstraintOp::EQ,
68  std::unique_ptr<AstArgument>(new AstVariable("X")),
69  std::unique_ptr<AstArgument>(new AstVariable("Y")));
70  clause->addToBody(std::unique_ptr<AstLiteral>(e1));
71 
72  // !b(Z)
73  auto* b = new AstAtom("b");
74  b->addArgument(std::unique_ptr<AstArgument>(new AstVariable("Z")));
75  auto* neg = new AstNegation(std::unique_ptr<AstAtom>(b));
76  clause->addToBody(std::unique_ptr<AstLiteral>(neg));
77 
78  // check construction
79  EXPECT_EQ("r(X,Y,Z) :- \n a(X),\n X = Y,\n !b(Z).", toString(*clause));
80 
81  auto program = std::make_unique<AstProgram>();
82  program->addClause(std::move(clause));
83  DebugReport dbgReport;
84  ErrorReport errReport;
85  AstTranslationUnit tu{std::move(program), errReport, dbgReport};
86 
87  // obtain groundness
88  auto isGrounded = getGroundedTerms(tu, *tu.getProgram()->getClauses()[0]);
89 
90  auto args = head->getArguments();
91  // check selected sub-terms
92  EXPECT_TRUE(isGrounded[args[0]]); // X
93  EXPECT_TRUE(isGrounded[args[1]]); // Y
94  EXPECT_FALSE(isGrounded[args[2]]); // Z
95 }

422 /**
423  * Test the removal of redundancies within clauses using the MinimiseProgramTransformer.
424  *
425  * In particular, the removal of:
426  * - intraclausal literals equivalent to another literal in the body
427  * e.g. a(x) :- b(x), b(x), c(x). --> a(x) :- b(x), c(x).
428  * - clauses that are only trivially satisfiable
429  * e.g. a(x) :- a(x), x != 0. is only true if a(x) is already true
430  */
431 TEST(AstUtils, RemoveClauseRedundancies) {
432  ErrorReport e;
433  DebugReport d;
434 
435  std::unique_ptr<AstTranslationUnit> tu = ParserDriver::parseTranslationUnit(
436  R"(
437  .decl a,b,c(X:number)
438  a(0).
439  b(1).
440  c(X) :- b(X).
441 
442  a(X) :- b(X), c(X).
443  a(X) :- a(X).
444  a(X) :- a(X), X != 1.
445 
446  q(X) :- a(X).
447 
448  .decl q(X:number)
449  .output q()
450  )",
451  e, d);
452 
453  const auto& program = *tu->getProgram();
454 
455  // Invoking the `RemoveRelationCopiesTransformer` to create some extra redundancy
456  // In particular: The relation `c` will be replaced with `b` throughout, creating
457  // the clause b(x) :- b(x).
458  std::make_unique<RemoveRelationCopiesTransformer>()->apply(*tu);
459  EXPECT_EQ(nullptr, getRelation(program, "c"));
460  auto bIntermediateClauses = getClauses(program, "b");
461  EXPECT_EQ(2, bIntermediateClauses.size());
462  EXPECT_EQ("b(1).", toString(*bIntermediateClauses[0]));
463  EXPECT_EQ("b(X) :- \n b(X).", toString(*bIntermediateClauses[1]));
464 
465  // Attempt to minimise the program
466  std::make_unique<MinimiseProgramTransformer>()->apply(*tu);
467  EXPECT_EQ(3, program.getRelations().size());
468 
469  auto aClauses = getClauses(program, "a");
470  EXPECT_EQ(2, aClauses.size());
471  EXPECT_EQ("a(0).", toString(*aClauses[0]));
472  EXPECT_EQ("a(X) :- \n b(X).", toString(*aClauses[1]));
473 
474  auto bClauses = getClauses(program, "b");
475  EXPECT_EQ(1, bClauses.size());
476  EXPECT_EQ("b(1).", toString(*bClauses[0]));
477 
478  auto qClauses = getClauses(program, "q");
479  EXPECT_EQ(1, qClauses.size());
480  EXPECT_EQ("q(X) :- \n a(X).", toString(*qClauses[0]));
481 }
