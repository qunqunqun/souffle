219 /**
220  * @class HoistConditionsTransformer
221  * @brief Hosts conditions in a loop-nest to the most-outer/semantically-correct loop
222  *
223  * Hoists the conditions to the earliest point in the loop nest where their
224  * evaluation is still semantically correct.
225  *
226  * The transformations assumes that filter operations are stored verbose,
227  * i.e. a conjunction is expressed by two consecutive filter operations.
228  * For example ..
229  *
230  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
231  * QUERY
232  * ...
233  * IF C1 /\ C2 then
234  * ...
235  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
236  *
237  * should be rewritten / or produced by the translator as
238  *
239  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
240  * QUERY
241  * ...
242  * IF C1
243  * IF C2
244  * ...
245  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
246  *
247  * otherwise the levelling becomes imprecise, i.e., for both conditions
248  * the most outer-level is sought rather than considered separately.
249  *
250  * If there are transformers prior to hoistConditions() that introduce
251  * conjunction, another transformer is required that splits the
252  * filter operations. However, at the moment this is not necessary
253  * because the translator delivers already the right RAM format.
254  *
255  * TODO: break-up conditions while transforming so that this requirement
256  * is removed.
257  */
258 class HoistConditionsTransformer : public RamTransformer {
259 public:
260  std::string getName() const override {
261  return "HoistConditionsTransformer";
262  }
263 
264  /**
265  * @brief Hoist filter operations.
266  * @param program that is transformed
267  * @return Flag showing whether the program has been changed by the transformation
268  *
269  * There are two types of conditions in
270  * filter operations. The first type depends on tuples of
271  * RamTupleOperation operations. The second type are independent of
272  * tuple access. Both types of conditions will be hoisted to
273  * the most out-scope such that the program is still valid.
274  */
275  bool hoistConditions(RamProgram& program);
276 
277 protected:
278  RamLevelAnalysis* rla{nullptr};
279 
280  bool transform(RamTranslationUnit& translationUnit) override {
281  rla = translationUnit.getAnalysis<RamLevelAnalysis>();
282  return hoistConditions(translationUnit.getProgram());
283  }

427 /**
428  * @class IndexedInequalityTransformer
429  * @brief Removes Inequalities from Indexed Operations and replaces them with a Filter Operation
430  * and empty Indexed Operations are coverted to their Non-Indexed semantic equivalent
431  *
432  * If there exists inequality constraints in an Indexed Operation, these constraints will be
433  * replaced with a semantically equivalent nested Filter Operation.
434  *
435  * Furthermore, if after removing all of these inequality constraints from the Indexed Operation
436  * we may find that the Indexed Operation is empty (no constraints).
437  * This occurs in the case where an Indexed Operation is composed entirely of inequality constraints.
438  * In this situation, the Indexed Operation is empty and replaced with a semantically equivalent Operation.
439  * i.e. RamIndexScan -> RamScan
440  *
441  * For example,
442  *
443  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
444  * QUERY
445  * ...
446  * FOR t1 IN X ON INDEX t1.x < 10 AND t1.y > 20
447  * ... // t1 only has inequality constraints placed on it
448  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
449  *
450  * will be rewritten to
451  *
452  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
453  * QUERY
454  * ...
455  * FOR t1 in X ON INDEX // empty index since all inequalities have been removed
456  * IF t1.x < 10 AND t1.y > 20
457  * // replaced with a semantically equivalent filter
458  * ...
459  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
460  *
461  * will be rewritten to
462  *
463  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
464  * QUERY
465  * ...
466  * FOR t1 in X // RamScan instead of RamIndexScan
467  * IF t1.x < 10 AND t1.y > 20
468  * // replaced with a semantically equivalent filter
469  * ...
470  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
471  *
472  */
