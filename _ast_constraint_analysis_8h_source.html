33 /**
34  * A variable type to be utilized by AST constraint analysis. Each such variable is
35  * associated with an AstArgument which's property it is describing.
36  *
37  * @tparam PropertySpace the property space associated to the analysis
38  */
39 template <typename PropertySpace>
40 struct AstConstraintAnalysisVar : public Variable<const AstArgument*, PropertySpace> {
41  explicit AstConstraintAnalysisVar(const AstArgument* arg)
42  : Variable<const AstArgument*, PropertySpace>(arg) {}
43  explicit AstConstraintAnalysisVar(const AstArgument& arg)
44  : Variable<const AstArgument*, PropertySpace>(&arg) {}
45 
46  /** adds print support */
47  void print(std::ostream& out) const override {
48  out << "var(" << *(this->id) << ")";
49  }
50 };

52 /**
53  * A base class for AstConstraintAnalysis collecting constraints for an analysis
54  * by visiting every node of a given AST. The collected constraints are
55  * then utilized to obtain the desired analysis result.
56  *
57  * @tparam AnalysisVar the type of variable (and included property space)
58  * to be utilized by this analysis.
59  */
60 template <typename AnalysisVar>
61 class AstConstraintAnalysis : public AstVisitor<void> {
62 public:
63  using value_type = typename AnalysisVar::property_space::value_type;
64  using constraint_type = std::shared_ptr<Constraint<AnalysisVar>>;
65  using solution_type = std::map<const AstArgument*, value_type>;
66 
67  virtual void collectConstraints(const AstClause& clause) {
68  visitDepthFirstPreOrder(clause, *this);
69  }
70 
71  /**
72  * Runs this constraint analysis on the given clause.
73  *
74  * @param clause the close to be analysed
75  * @param debug a flag enabling the printing of debug information
76  * @return an assignment mapping a property to each argument in the given clause
77  */
78  solution_type analyse(const AstClause& clause, std::ostream* debugOutput = nullptr) {
79  collectConstraints(clause);
80 
81  assignment = constraints.solve();
82 
83  // print debug information if desired
84  if (debugOutput != nullptr) {
85  *debugOutput << "Clause: " << clause << "\n";
86  *debugOutput << "Problem:\n" << constraints << "\n";
87  *debugOutput << "Solution:\n" << assignment << "\n";
88  }
89 
90  // convert assignment to result
91  solution_type solution;
92  visitDepthFirst(clause, [&](const AstArgument& arg) { solution[&arg] = assignment[getVar(arg)]; });
93  return solution;
94  }
95 
96 protected:
97  /**
98  * A utility function mapping an AstArgument to its associated analysis variable.
99  *
100  * @param arg the AST argument to be mapped
101  * @return the analysis variable representing its associated value
102  */
103  AnalysisVar getVar(const AstArgument& arg) {
104  const auto* var = dynamic_cast<const AstVariable*>(&arg);
105  if (var == nullptr) {
106  // no mapping required
107  return AnalysisVar(arg);
108  }
109 
110  // filter through map => always take the same variable
111  auto res = variables.insert({var->getName(), AnalysisVar(var)}).first;
112  return res->second;
113  }
114 
115  /**
116  * A utility function mapping an AstArgument to its associated analysis variable.
117  *
118  * @param arg the AST argument to be mapped
119  * @return the analysis variable representing its associated value
120  */
121  AnalysisVar getVar(const AstArgument* arg) {
122  return getVar(*arg);
123  }
124 
125  /** Adds another constraint to the internally maintained list of constraints */
126  void addConstraint(const constraint_type& constraint) {
127  constraints.add(constraint);
128  }
129 
130  Assignment<AnalysisVar> assignment;
131 
132  /** The list of constraints making underlying this analysis */
133  Problem<AnalysisVar> constraints;
134 
135  /** A map mapping variables to unique instances to facilitate the unification of variables */
136  std::map<std::string, AnalysisVar> variables;
137 };
