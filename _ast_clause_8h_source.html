74  */
75 class AstExecutionPlan : public AstNode {
76 public:
77  /** updates execution order for rule version */
78  void setOrderFor(int version, Own<AstExecutionOrder> plan) {
79  plans[version] = std::move(plan);
80  }
81 
82  /** get orders */
83  std::map<int, const AstExecutionOrder*> getOrders() const {
84  std::map<int, const AstExecutionOrder*> result;
85  for (auto& plan : plans) {
86  result.insert(std::make_pair(plan.first, plan.second.get()));
87  }
88  return result;
89  }
90 
91  AstExecutionPlan* clone() const override {
92  auto res = new AstExecutionPlan();
93  res->setSrcLoc(getSrcLoc());
94  for (auto& plan : plans) {
95  res->setOrderFor(plan.first, Own<AstExecutionOrder>(plan.second->clone()));
96  }
97  return res;
98  }
99 
100  void apply(const AstNodeMapper& map) override {
101  for (auto& plan : plans) {
102  plan.second = map(std::move(plan.second));
103  }
104  }
105 
106  std::vector<const AstNode*> getChildNodes() const override {
107  std::vector<const AstNode*> childNodes;
108  for (auto& plan : plans) {
109  childNodes.push_back(plan.second.get());
110  }
111  return childNodes;
112  }
113 
114 protected:
115  void print(std::ostream& out) const override {
116  if (!plans.empty()) {
117  out << " .plan ";
118  out << join(plans, ", ",
119  [](std::ostream& os, const auto& arg) { os << arg.first << ":" << *arg.second; });
120  }
121  }
122 
123  bool equal(const AstNode& node) const override {
124  const auto& other = static_cast<const AstExecutionPlan&>(node);
125  return equal_targets(plans, other.plans);
126  }
127 
128 private:
129  /** mapping versions of clauses to execution plans */
130  std::map<int, Own<AstExecutionOrder>> plans;
131 };

133 /**
134  * Intermediate representation of a datalog clause.
135  *
136  * A clause can either be:
137  * - a fact - a clause with no body (e.g., X(a,b))
138  * - a rule - a clause with a head and a body (e.g., Y(a,b) -: X(a,b))
139  *
140  * TODO (azreika): make clause abstract and split into two subclasses: Rule and Fact
141  */
142 class AstClause : public AstNode {
143 public:
144  AstClause(Own<AstAtom> head = {}, VecOwn<AstLiteral> bodyLiterals = {}, Own<AstExecutionPlan> plan = {},
145  SrcLocation loc = {})
146  : AstNode(std::move(loc)), head(std::move(head)), bodyLiterals(std::move(bodyLiterals)),
147  plan(std::move(plan)) {}
148 
149  /** Add a Literal to the body of the clause */
150  void addToBody(Own<AstLiteral> literal) {
151  bodyLiterals.push_back(std::move(literal));
152  }
153 
154  /** Set the head of clause to @p h */
155  void setHead(Own<AstAtom> h) {
156  head = std::move(h);
157  }
158 
159  /** Set the bodyLiterals of clause to @p body */
160  void setBodyLiterals(VecOwn<AstLiteral> body) {
161  bodyLiterals = std::move(body);
162  }
163 
164  /** Return the atom that represents the head of the clause */
165  AstAtom* getHead() const {
166  return head.get();
167  }
168 
169  /** Obtains a copy of the internally maintained body literals */
170  std::vector<AstLiteral*> getBodyLiterals() const {
171  return toPtrVector(bodyLiterals);
172  }
173 
174  /** Obtains the execution plan associated to this clause or null if there is none */
175  const AstExecutionPlan* getExecutionPlan() const {
176  return plan.get();
177  }
178 
179  /** Updates the execution plan associated to this clause */
180  void setExecutionPlan(Own<AstExecutionPlan> plan) {
181  this->plan = std::move(plan);
182  }
183 
184  /** Resets the execution plan */
185  void clearExecutionPlan() {
186  plan = nullptr;
187  }
188 
189  AstClause* clone() const override {
190  return new AstClause(
191  souffle::clone(head), souffle::clone(bodyLiterals), souffle::clone(plan), getSrcLoc());
192  }
193 
194  void apply(const AstNodeMapper& map) override {
195  head = map(std::move(head));
196  for (auto& lit : bodyLiterals) {
197  lit = map(std::move(lit));
198  }
199  }
200 
201  std::vector<const AstNode*> getChildNodes() const override {
202  std::vector<const AstNode*> res = {head.get()};
203  for (auto& cur : bodyLiterals) {
204  res.push_back(cur.get());
205  }
206  return res;
207  }
208 
209 protected:
210  void print(std::ostream& os) const override {
211  if (head != nullptr) {
212  os << *head;
213  }
214  if (!bodyLiterals.empty()) {
215  os << " :- \n " << join(bodyLiterals, ",\n ");
216  }
217  os << ".";
218  if (plan != nullptr) {
219  os << *plan;
220  }
221  }
222 
223  bool equal(const AstNode& node) const override {
224  const auto& other = static_cast<const AstClause&>(node);
225  return equal_ptr(head, other.head) && equal_targets(bodyLiterals, other.bodyLiterals) &&
226  equal_ptr(plan, other.plan);
227  }
228 
229  /** head of the clause */
230  Own<AstAtom> head;
231 
232  /** body literals of clause */
233  VecOwn<AstLiteral> bodyLiterals;
234 
235  /** user defined execution plan (if not defined, plan is null) */
236  Own<AstExecutionPlan> plan;
237 };
