28  * A simple graph structure for graph-based operations.
29  */
30 template <typename Vertex, typename Compare = std::less<Vertex>>
31 class Graph {
32 public:
33  /**

35  */

74  /** Determines whether the given edge is present */
75  bool contains(const Vertex& from, const Vertex& to) const {
76  auto pos = _successors.find(from);
77  if (pos == _successors.end()) {
78  return false;
79  }
80  auto p2 = pos->second.find(to);
81  return p2 != pos->second.end();
82  }

84  /** Determines whether there is a directed path between the two vertices */
85  bool reaches(const Vertex& from, const Vertex& to) const {
86  // quick check
87  if (!contains(from) || !contains(to)) {
88  return false;
89  }
90 
91  // conduct a depth-first search starting at from
92  bool found = false;
93  bool first = true;
94  visitDepthFirst(from, [&](const Vertex& cur) {
95  found = !first && (found || cur == to);
96  first = false;
97  });
98  return found;
99  }
100 
101  /** Obtains the set of all vertices in the same clique than the given vertex */
102  const std::set<Vertex, Compare> clique(const Vertex& vertex) const {
103  std::set<Vertex, Compare> res;
104  res.insert(vertex);
105  for (const auto& cur : vertices()) {
106  if (reaches(vertex, cur) && reaches(cur, vertex)) {
107  res.insert(cur);
108  }
109  }
110  return res;
111  }

113  /** A generic utility for depth-first visits */
114  template <typename Lambda>
115  void visitDepthFirst(const Vertex& vertex, const Lambda& lambda) const {
116  std::set<Vertex, Compare> visited;
117  visitDepthFirst(vertex, lambda, visited);
118  }

121  void print(std::ostream& out) const {

123  out << "{";

141 private:

147  /** The internal implementation of depth-first visits */
148  template <typename Lambda>
149  void visitDepthFirst(
150  const Vertex& vertex, const Lambda& lambda, std::set<Vertex, Compare>& visited) const {
151  lambda(vertex);
152  auto pos = _successors.find(vertex);
153  if (pos == _successors.end()) {
154  return;
155  }
156  for (const auto& cur : pos->second) {
157  if (visited.insert(cur).second) {
158  visitDepthFirst(cur, lambda, visited);
159  }
160  }
161  }
