57  void writeNextTupleElement(std::ostream& destination, const std::string& name, const RamDomain value) {
58  using ValueTuple = std::pair<const std::string, const RamDomain>;
59  std::stack<std::variant<ValueTuple, std::string>> worklist;
60  worklist.push(std::make_pair(name, value));
61 
62  // the Json11 output is not tail recursive, therefore highly inefficient for recursive record
63  // in addition the JSON object is immutable, so has memory overhead
64  while (!worklist.empty()) {
65  std::variant<ValueTuple, std::string> curr = worklist.top();
66  worklist.pop();
67 
68  if (std::holds_alternative<std::string>(curr)) {
69  destination << std::get<std::string>(curr);
70  continue;
71  }
72 
73  const std::string& currType = std::get<ValueTuple>(curr).first;
74  const RamDomain currValue = std::get<ValueTuple>(curr).second;
75  assert(currType.length() > 2 && "Invalid type length");
76  switch (currType[0]) {
77  // since some strings may need to be escaped, we use dump here
78  case 's': destination << Json(symbolTable.unsafeResolve(currValue)).dump(); break;
79  case 'i': destination << currValue; break;
80  case 'u': destination << (int)ramBitCast<RamUnsigned>(currValue); break;
81  case 'f': destination << ramBitCast<RamFloat>(currValue); break;
82  case 'r': {
83  auto&& recordInfo = types["records"][currType];
84  assert(!recordInfo.is_null() && "Missing record type information");
85  if (currValue == 0) {
86  destination << "null";
87  break;
88  }
89 
90  auto&& recordTypes = recordInfo["types"];
91  const size_t recordArity = recordInfo["arity"].long_value();
92  const RamDomain* tuplePtr = recordTable.unpack(currValue, recordArity);
93  worklist.push("]");
94  for (auto i = (long long)(recordArity - 1); i >= 0; --i) {
95  if (i != (long long)(recordArity - 1)) {
96  worklist.push(", ");
97  }
98  const std::string& recordType = recordTypes[i].string_value();
99  const RamDomain recordValue = tuplePtr[i];
100  worklist.push(std::make_pair(recordType, recordValue));
101  }
102 
103  worklist.push("[");
104  break;
105  }
106  default: fatal("unsupported type attribute: `%c`", currType[0]);
107  }
108  }
109  }
