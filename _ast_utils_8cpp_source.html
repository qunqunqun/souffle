116 std::set<const AstRelation*> getBodyRelations(const AstClause* clause, const AstProgram* program) {
117  std::set<const AstRelation*> bodyRelations;
118  for (const auto& lit : clause->getBodyLiterals()) {
119  visitDepthFirst(
120  *lit, [&](const AstAtom& atom) { bodyRelations.insert(getAtomRelation(&atom, program)); });
121  }
122  for (const auto& arg : clause->getHead()->getArguments()) {

125  }

129 size_t getClauseNum(const AstProgram* program, const AstClause* clause) {
130  // TODO (azreika): This number might change between the provenance transformer and the AST->RAM
131  // translation. Might need a better way to assign IDs to clauses... (see PR #1288).
132  const AstRelation* rel = getRelation(*program, clause->getHead()->getQualifiedName());
133  assert(rel != nullptr && "clause relation does not exist");
134 
135  size_t clauseNum = 1;
136  for (const auto* cur : getClauses(*program, *rel)) {
137  bool isFact = cur->getBodyLiterals().empty();
138  if (cur == clause) {
139  return isFact ? 0 : clauseNum;
140  }
141 
142  if (!isFact) {
143  clauseNum++;
144  }
145  }
146 
147  fatal("clause does not exist");

193 bool isFact(const AstClause& clause) {
194  // there must be a head
195  if (clause.getHead() == nullptr) {
196  return false;
197  }
198  // there must not be any body clauses
199  if (!clause.getBodyLiterals().empty()) {
200  return false;
201  }
202 
203  // and there are no aggregates
204  bool hasAggregatesOrMultiResultFunctor = false;
205  visitDepthFirst(*clause.getHead(), [&](const AstArgument& arg) {
206  if (dynamic_cast<const AstAggregator*>(&arg)) {
207  hasAggregatesOrMultiResultFunctor = true;
208  }
209 
210  auto func = dynamic_cast<const AstIntrinsicFunctor*>(&arg);
211  auto info = func ? func->getFunctionInfo() : nullptr;
212  if (info && info->multipleResults) {
213  hasAggregatesOrMultiResultFunctor = true;
214  }
215  });
216  return !hasAggregatesOrMultiResultFunctor;
217 }

233 AstClause* reorderAtoms(const AstClause* clause, const std::vector<unsigned int>& newOrder) {
234  // Find all atom positions
235  std::vector<unsigned int> atomPositions;
236  std::vector<AstLiteral*> bodyLiterals = clause->getBodyLiterals();
237  for (unsigned int i = 0; i < bodyLiterals.size(); i++) {
238  if (dynamic_cast<AstAtom*>(bodyLiterals[i]) != nullptr) {
239  atomPositions.push_back(i);
240  }
241  }
242 
243  // Validate given order
244  assert(newOrder.size() == atomPositions.size());
245  std::vector<unsigned int> nopOrder;
246  for (unsigned int i = 0; i < atomPositions.size(); i++) {
247  nopOrder.push_back(i);
248  }
249  assert(std::is_permutation(nopOrder.begin(), nopOrder.end(), newOrder.begin()));
250 
251  // Create a new clause with the given atom order, leaving the rest unchanged
252  AstClause* newClause = cloneHead(clause);
253  unsigned int currentAtom = 0;
254  for (unsigned int currentLiteral = 0; currentLiteral < bodyLiterals.size(); currentLiteral++) {
255  AstLiteral* literalToAdd = bodyLiterals[currentLiteral];
256  if (dynamic_cast<AstAtom*>(literalToAdd) != nullptr) {
257  // Atoms should be reordered
258  literalToAdd = bodyLiterals[atomPositions[newOrder[currentAtom++]]];
259  }
260  newClause->addToBody(souffle::clone(literalToAdd));
261  }
262 
263  return newClause;
264 }

276 IntrinsicFunctors validOverloads(const TypeAnalysis& typing, const AstIntrinsicFunctor& func) {
277  auto typeAttrs = [&](const AstArgument* arg) -> std::set<TypeAttribute> {
278  auto&& types = typing.getTypes(arg);
279  if (types.isAll())
280  return {TypeAttribute::Signed, TypeAttribute::Unsigned, TypeAttribute::Float,
281  TypeAttribute::Symbol, TypeAttribute::Record};
282 
283  std::set<TypeAttribute> tyAttrs;
284  for (auto&& ty : types)
285  tyAttrs.insert(getTypeAttribute(ty));
286  return tyAttrs;
287  };
288  auto retTys = typeAttrs(&func);
289  auto argTys = map(func.getArguments(), typeAttrs);
290 
291  auto candidates = filterNot(functorBuiltIn(func.getFunction()), [&](const IntrinsicFunctor& x) -> bool {
292  if (!x.variadic && argTys.size() != x.params.size()) return true; // arity mismatch?
293 
294  for (size_t i = 0; i < argTys.size(); ++i)
295  if (!contains(argTys[i], x.params[x.variadic ? 0 : i])) return true;
296 
297  return !contains(retTys, x.result);
298  });
299 
300  std::sort(
301  candidates.begin(), candidates.end(), [&](const IntrinsicFunctor& a, const IntrinsicFunctor& b) {
302  if (a.result != b.result) return a.result < b.result;
303  if (a.variadic != b.variadic) return a.variadic < b.variadic;
304  return std::lexicographical_compare(
305  a.params.begin(), a.params.end(), b.params.begin(), b.params.end());
306  });
307  return candidates;
308 }
