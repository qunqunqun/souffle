110  /**
111  * Abstract replacement method for a node.
112  *
113  * If the given nodes is to be replaced, the handed in node

116  */
117  virtual Own<AstNode> operator()(Own<AstNode> node) const = 0;
118 
119  /**
120  * Wrapper for any subclass of the AST node hierarchy performing type casts.
121  */
122  template <typename T>
123  Own<T> operator()(Own<T> node) const {
124  Own<AstNode> resPtr = (*this)(Own<AstNode>(static_cast<AstNode*>(node.release())));
125  assert(nullptr != dynamic_cast<T*>(resPtr.get()) && "Invalid target node!");
126  return Own<T>(dynamic_cast<T*>(resPtr.release()));
127  }
