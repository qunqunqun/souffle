93 bool RemoveRelationCopiesTransformer::removeRelationCopies(AstTranslationUnit& translationUnit) {
94  using alias_map = std::map<AstQualifiedName, AstQualifiedName>;
95 
96  // collect aliases
97  alias_map isDirectAliasOf;
98 
99  auto* ioType = translationUnit.getAnalysis<IOType>();
100 
101  AstProgram& program = *translationUnit.getProgram();
102 
103  // search for relations only defined by a single rule ..
104  for (AstRelation* rel : program.getRelations()) {
105  const auto& clauses = getClauses(program, *rel);
106  if (!ioType->isIO(rel) && clauses.size() == 1u) {
107  // .. of shape r(x,y,..) :- s(x,y,..)
108  AstClause* cl = clauses[0];
109  std::vector<AstAtom*> bodyAtoms = getBodyLiterals<AstAtom>(*cl);
110  if (!isFact(*cl) && cl->getBodyLiterals().size() == 1u && bodyAtoms.size() == 1u) {
111  AstAtom* atom = bodyAtoms[0];
112  if (equal_targets(cl->getHead()->getArguments(), atom->getArguments())) {
113  // Requirements:
114  // 1) (checked) It is a rule with exactly one body.
115  // 3) (checked) The body consists of an atom.
116  // 4) (checked) The atom's arguments must be identical to the rule's head.
117  // 5) (pending) The rules's head must consist only of either:
118  // 5a) Variables
119  // 5b) Records unpacked into variables
120  // 6) (pending) Each variable must have a distinct name.
121  bool onlyDistinctHeadVars = true;
122  std::set<std::string> headVars;
123 
124  auto args = cl->getHead()->getArguments();
125  while (onlyDistinctHeadVars && !args.empty()) {
126  const auto cur = args.back();
127  args.pop_back();
128 
129  if (auto var = dynamic_cast<const AstVariable*>(cur)) {
130  onlyDistinctHeadVars &= headVars.insert(var->getName()).second;
131  } else if (auto init = dynamic_cast<const AstRecordInit*>(cur)) {
132  // records are decomposed and their arguments are checked
133  for (auto rec_arg : init->getArguments()) {
134  args.push_back(rec_arg);
135  }
136  } else {
137  onlyDistinctHeadVars = false;
138  }
139  }
140 
141  if (onlyDistinctHeadVars) {
142  // all arguments are either distinct variables or records unpacked into distinct
143  // variables
144  isDirectAliasOf[cl->getHead()->getQualifiedName()] = atom->getQualifiedName();
145  }
146  }
147  }
148  }
149  }
150 
151  // map each relation to its ultimate alias (could be transitive)
152  alias_map isAliasOf;
153 
154  // track any copy cycles; cyclic rules are effectively empty
155  std::set<AstQualifiedName> cycle_reps;
156 
157  for (std::pair<AstQualifiedName, AstQualifiedName> cur : isDirectAliasOf) {
158  // compute replacement
159 
160  std::set<AstQualifiedName> visited;
161  visited.insert(cur.first);
162  visited.insert(cur.second);
163 
164  auto pos = isDirectAliasOf.find(cur.second);
165  while (pos != isDirectAliasOf.end()) {
166  if (visited.count(pos->second) != 0u) {
167  cycle_reps.insert(cur.second);
168  break;
169  }
170  cur.second = pos->second;
171  pos = isDirectAliasOf.find(cur.second);
172  }
173  isAliasOf[cur.first] = cur.second;
174  }
175 
176  if (isAliasOf.empty()) {
177  return false;
178  }
179 
180  // replace usage of relations according to alias map
181  visitDepthFirst(program, [&](const AstAtom& atom) {
182  auto pos = isAliasOf.find(atom.getQualifiedName());
183  if (pos != isAliasOf.end()) {
184  const_cast<AstAtom&>(atom).setQualifiedName(pos->second);
185  }
186  });
187 
188  // break remaining cycles
189  for (const auto& rep : cycle_reps) {
190  const auto& rel = *getRelation(program, rep);
191  const auto& clauses = getClauses(program, rel);
192  assert(clauses.size() == 1u && "unexpected number of clauses in relation");
193  program.removeClause(clauses[0]);
194  }
195 
196  // remove unused relations
197  for (const auto& cur : isAliasOf) {
198  if (cycle_reps.count(cur.first) == 0u) {
199  program.removeRelation(getRelation(program, cur.first)->getQualifiedName());
200  }
201  }
202 
203  return true;
204 }

259 bool MaterializeSingletonAggregationTransformer::transform(AstTranslationUnit& translationUnit) {
260  AstProgram& program = *translationUnit.getProgram();
261  std::set<std::pair<AstAggregator*, AstClause*>> pairs;
262  // collect references to clause / aggregate pairs
263  visitDepthFirst(program, [&](const AstClause& clause) {
264  visitDepthFirst(clause, [&](const AstAggregator& agg) {
265  // if the aggregate isn't single valued
266  // (ie it relies on a grounding from the outer scope)
267  // or it's the only atom in the clause, then there's no point materialising it!
268  if (!isSingleValued(agg, clause) || clause.getBodyLiterals().size() == 1) {
269  return;
270  }
271  auto* foundAggregate = const_cast<AstAggregator*>(&agg);
272  auto* foundClause = const_cast<AstClause*>(&clause);
273  pairs.insert(std::make_pair(foundAggregate, foundClause));
274  });
275  });
276  for (auto pair : pairs) {
277  // Clone the aggregate that we're going to be deleting.
278  auto aggregate = souffle::clone(pair.first);
279  AstClause* clause = pair.second;
280  // synthesise an aggregate relation
281  // __agg_rel_0()
282  auto aggRel = std::make_unique<AstRelation>();
283  auto aggHead = std::make_unique<AstAtom>();
284  auto aggClause = std::make_unique<AstClause>();
285 
286  std::string aggRelName = findUniqueAggregateRelationName(program);
287  aggRel->setQualifiedName(aggRelName);
288  aggHead->setQualifiedName(aggRelName);
289 
290  // create a synthesised variable to replace the aggregate term!
291  std::string variableName = findUniqueVariableName(*clause);
292  auto variable = std::make_unique<AstVariable>(variableName);
293 
294  // __agg_rel_0(z) :- ...
295  aggHead->addArgument(souffle::clone(variable));
296  aggRel->addAttribute(std::make_unique<AstAttribute>(variableName, "number"));
297  aggClause->setHead(souffle::clone(aggHead));
298 
299  // A(x) :- x = sum .., B(x).
300  // -> A(x) :- x = z, B(x), __agg_rel_0(z).
301  auto equalityLiteral = std::make_unique<AstBinaryConstraint>(
302  BinaryConstraintOp::EQ, souffle::clone(variable), souffle::clone(aggregate));
303  // __agg_rel_0(z) :- z = sum ...
304  aggClause->addToBody(std::move(equalityLiteral));
305  program.addRelation(std::move(aggRel));
306  program.addClause(std::move(aggClause));
307 
308  // the only thing left to do is just replace the aggregate terms in the original
309  // clause with the synthesised variable
310  struct replaceAggregate : public AstNodeMapper {
311  const AstAggregator& aggregate;
312  const std::unique_ptr<AstVariable> variable;
313  replaceAggregate(const AstAggregator& aggregate, std::unique_ptr<AstVariable> variable)
314  : aggregate(aggregate), variable(std::move(variable)) {}
315  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
316  assert(node != nullptr);
317  if (auto* current = dynamic_cast<AstAggregator*>(node.get())) {
318  if (*current == aggregate) {
319  auto replacement = souffle::clone(variable);
320  assert(replacement != nullptr);
321  return replacement;
322  }
323  }
324  node->apply(*this);
325  assert(node != nullptr);
326  return node;
327  }
328  };
329  replaceAggregate update(*aggregate, std::move(variable));
330  clause->apply(update);
331  clause->addToBody(std::move(aggHead));
332  }
333  return pairs.size() > 0;
334 }

358 bool MaterializeAggregationQueriesTransformer::materializeAggregationQueries(
359  AstTranslationUnit& translationUnit) {
360  bool changed = false;
361 
362  AstProgram& program = *translationUnit.getProgram();
363  const TypeEnvironment& env = translationUnit.getAnalysis<TypeEnvironmentAnalysis>()->getTypeEnvironment();
364 
365  // if an aggregator has a body consisting of more than an atom => create new relation
366  int counter = 0;
367  visitDepthFirst(program, [&](const AstClause& clause) {
368  visitDepthFirst(clause, [&](const AstAggregator& agg) {
369  // check whether a materialization is required
370  if (!needsMaterializedRelation(agg)) {
371  return;
372  }
373  changed = true;
374 
375  // -- create a new clause --
376 
377  auto relName = "__agg_body_rel_" + toString(counter++);
378  while (getRelation(program, relName) != nullptr) {
379  relName = "__agg_body_rel_" + toString(counter++);
380  }
381  // create the new clause for the materialised rule
382  auto* aggClause = new AstClause();
383  // create the body of the new materialised rule
384  for (const auto& cur : agg.getBodyLiterals()) {
385  aggClause->addToBody(souffle::clone(cur));
386  }
387  // find stuff for which we need a grounding
388  for (const auto& argPair : getGroundedTerms(translationUnit, *aggClause)) {
389  const auto* variable = dynamic_cast<const AstVariable*>(argPair.first);
390  bool variableIsGrounded = argPair.second;
391  // if it's not even a variable type or the term is grounded
392  // then skip it
393  if (variable == nullptr || variableIsGrounded) {
394  continue;
395  }
396 
397  for (const auto& lit : clause.getBodyLiterals()) {
398  const auto* atom = dynamic_cast<const AstAtom*>(lit);
399  if (atom == nullptr) {
400  continue; // it's not an atom so it can't help ground anything
401  }
402  for (const auto& arg : atom->getArguments()) {
403  const auto* atomVariable = dynamic_cast<const AstVariable*>(arg);
404  // if this atom contains the variable I need to ground, add it
405  if ((atomVariable != nullptr) && variable->getName() == atomVariable->getName()) {
406  // expand the body with this one so that it will ground this variable
407  aggClause->addToBody(souffle::clone(atom));
408  break;
409  }
410  }
411  }
412  }
413 
414  auto* head = new AstAtom();
415  head->setQualifiedName(relName);
416  std::vector<bool> symbolArguments;
417 
418  // Ensure each variable is only added once
419  std::set<std::string> variables;
420  visitDepthFirst(*aggClause, [&](const AstVariable& var) { variables.insert(var.getName()); });
421 
422  // Insert all variables occurring in the body of the aggregate into the head
423  for (const auto& var : variables) {
424  head->addArgument(std::make_unique<AstVariable>(var));
425  }
426 
427  aggClause->setHead(std::unique_ptr<AstAtom>(head));
428 
429  // instantiate unnamed variables in count operations
430  if (agg.getOperator() == AggregateOp::COUNT) {
431  int count = 0;
432  for (const auto& cur : aggClause->getBodyLiterals()) {
433  cur->apply(makeLambdaAstMapper(
434  [&](std::unique_ptr<AstNode> node) -> std::unique_ptr<AstNode> {
435  // check whether it is a unnamed variable
436  auto* var = dynamic_cast<AstUnnamedVariable*>(node.get());
437  if (var == nullptr) {
438  return node;
439  }
440 
441  // replace by variable
442  auto name = " _" + toString(count++);
443  auto res = new AstVariable(name);
444 
445  // extend head
446  head->addArgument(souffle::clone(res));
447 
448  // return replacement
449  return std::unique_ptr<AstNode>(res);
450  }));
451  }
452  }
453 
454  // -- build relation --
455 
456  auto* rel = new AstRelation();
457  rel->setQualifiedName(relName);
458  // add attributes
459  std::map<const AstArgument*, TypeSet> argTypes =
460  TypeAnalysis::analyseTypes(env, *aggClause, program);
461  for (const auto& cur : head->getArguments()) {
462  rel->addAttribute(std::make_unique<AstAttribute>(toString(*cur),
463  (isOfKind(argTypes[cur], TypeAttribute::Signed)) ? "number" : "symbol"));
464  }
465 
466  program.addClause(std::unique_ptr<AstClause>(aggClause));
467  program.addRelation(std::unique_ptr<AstRelation>(rel));
468 
469  // -- update aggregate --
470 
471  // count the usage of variables in the clause
472  // outside of aggregates. Note that the visitor
473  // is exhaustive hence double counting occurs
474  // which needs to be deducted for variables inside
475  // the aggregators and variables in the expression
476  // of aggregate need to be added. Counter is zero
477  // if the variable is local to the aggregate
478  std::map<std::string, int> varCtr;
479  visitDepthFirst(clause, [&](const AstArgument& arg) {
480  if (const auto* a = dynamic_cast<const AstAggregator*>(&arg)) {
481  visitDepthFirst(arg, [&](const AstVariable& var) { varCtr[var.getName()]--; });
482  if (a->getTargetExpression() != nullptr) {
483  visitDepthFirst(*a->getTargetExpression(),
484  [&](const AstVariable& var) { varCtr[var.getName()]++; });
485  }
486  } else {
487  visitDepthFirst(arg, [&](const AstVariable& var) { varCtr[var.getName()]++; });
488  }
489  });
490  std::vector<std::unique_ptr<AstArgument>> args;
491  for (auto arg : head->getArguments()) {
492  if (auto* var = dynamic_cast<AstVariable*>(arg)) {
493  // replace local variable by underscore if local
494  if (varCtr[var->getName()] == 0) {
495  args.emplace_back(new AstUnnamedVariable());
496  continue;
497  }
498  }
499  args.emplace_back(arg->clone());
500  }
501  auto aggAtom =
502  std::make_unique<AstAtom>(head->getQualifiedName(), std::move(args), head->getSrcLoc());
503 
504  std::vector<std::unique_ptr<AstLiteral>> newBody;
505  newBody.push_back(std::move(aggAtom));
506  const_cast<AstAggregator&>(agg).setBody(std::move(newBody));
507  });
508  });
509  return changed;
510 }

512 bool MaterializeAggregationQueriesTransformer::needsMaterializedRelation(const AstAggregator& agg) {
513  // everything with more than 1 body literal => materialize
514  if (agg.getBodyLiterals().size() > 1) {
515  return true;
516  }
517 
518  // Inspect remaining atom more closely
519  const AstAtom* atom = dynamic_cast<const AstAtom*>(agg.getBodyLiterals()[0]);
520  if (atom == nullptr) {
521  // No atoms, so materialize
522  return true;
523  }
524 
525  // If the same variable occurs several times => materialize
526  bool duplicates = false;
527  std::set<std::string> vars;
528  visitDepthFirst(*atom,
529  [&](const AstVariable& var) { duplicates = duplicates || !vars.insert(var.getName()).second; });
530 
531  // If there are duplicates a materialization is required
532  // for all others the materialization can be skipped
533  return duplicates;
534 }

565 bool RemoveEmptyRelationsTransformer::removeEmptyRelationUses(
566  AstTranslationUnit& translationUnit, AstRelation* emptyRelation) {
567  AstProgram& program = *translationUnit.getProgram();
568  bool changed = false;
569 
570  //
571  // (1) drop rules from the program that have empty relations in their bodies.
572  // (2) drop negations of empty relations
573  //
574  // get all clauses
575  std::vector<const AstClause*> clauses;
576  visitDepthFirst(program, [&](const AstClause& cur) { clauses.push_back(&cur); });
577 
578  // clean all clauses
579  for (const AstClause* cl : clauses) {
580  // check for an atom whose relation is the empty relation
581 
582  bool removed = false;
583  for (AstLiteral* lit : cl->getBodyLiterals()) {
584  if (auto* arg = dynamic_cast<AstAtom*>(lit)) {
585  if (getAtomRelation(arg, &program) == emptyRelation) {
586  program.removeClause(cl);
587  removed = true;
588  changed = true;
589  break;
590  }
591  }
592  }
593 
594  if (!removed) {
595  // check whether a negation with empty relations exists
596 
597  bool rewrite = false;
598  for (AstLiteral* lit : cl->getBodyLiterals()) {
599  if (auto* neg = dynamic_cast<AstNegation*>(lit)) {
600  if (getAtomRelation(neg->getAtom(), &program) == emptyRelation) {
601  rewrite = true;
602  break;
603  }
604  }
605  }
606 
607  if (rewrite) {
608  // clone clause without negation for empty relations
609 
610  auto res = std::unique_ptr<AstClause>(cloneHead(cl));
611 
612  for (AstLiteral* lit : cl->getBodyLiterals()) {
613  if (auto* neg = dynamic_cast<AstNegation*>(lit)) {
614  if (getAtomRelation(neg->getAtom(), &program) != emptyRelation) {
615  res->addToBody(souffle::clone(lit));
616  }
617  } else {
618  res->addToBody(souffle::clone(lit));
619  }
620  }
621 
622  program.removeClause(cl);
623  program.addClause(std::move(res));
624  changed = true;
625  }
626  }
627  }
628 
629  return changed;
630 }

757 bool PartitionBodyLiteralsTransformer::transform(AstTranslationUnit& translationUnit) {
758  bool changed = false;
759  AstProgram& program = *translationUnit.getProgram();
760 
761  /* Process:
762  * Go through each clause and construct a variable dependency graph G.
763  * The nodes of G are the variables. A path between a and b exists iff
764  * a and b appear in a common body literal.
765  *
766  * Using the graph, we can extract the body literals that are not associated
767  * with the arguments in the head atom into new relations. Depending on
768  * variable dependencies among these body literals, the literals can
769  * be partitioned into multiple separate new propositional clauses.
770  *
771  * E.g. a(x) <- b(x), c(y), d(y), e(z), f(z). can be transformed into:
772  * - a(x) <- b(x), newrel0(), newrel1().
773  * - newrel0() <- c(y), d(y).
774  * - newrel1() <- e(z), f(z).
775  *
776  * Note that only one pass through the clauses is needed:
777  * - All arguments in the body literals of the transformed clause cannot be
778  * independent of the head arguments (by construction)
779  * - The new relations holding the disconnected body literals are propositional,
780  * hence having no head arguments, and so the transformation does not apply.
781  */
782 
783  // Store clauses to add and remove after analysing the program
784  std::vector<AstClause*> clausesToAdd;
785  std::vector<const AstClause*> clausesToRemove;
786 
787  // The transformation is local to each rule, so can visit each independently
788  visitDepthFirst(program, [&](const AstClause& clause) {
789  // Create the variable dependency graph G
790  Graph<std::string> variableGraph = Graph<std::string>();
791  std::set<std::string> ruleVariables;
792 
793  // Add in the nodes
794  // The nodes of G are the variables in the rule
795  visitDepthFirst(clause, [&](const AstVariable& var) {
796  variableGraph.insert(var.getName());
797  ruleVariables.insert(var.getName());
798  });
799 
800  // Add in the edges
801  // Since we are only looking at reachability in the final graph, it is
802  // enough to just add in an (undirected) edge from the first variable
803  // in the literal to each of the other variables.
804  std::vector<AstLiteral*> literalsToConsider = clause.getBodyLiterals();
805  literalsToConsider.push_back(clause.getHead());
806 
807  for (AstLiteral* clauseLiteral : literalsToConsider) {
808  std::set<std::string> literalVariables;
809 
810  // Store all variables in the literal
811  visitDepthFirst(
812  *clauseLiteral, [&](const AstVariable& var) { literalVariables.insert(var.getName()); });
813 
814  // No new edges if only one variable is present
815  if (literalVariables.size() > 1) {
816  std::string firstVariable = *literalVariables.begin();
817  literalVariables.erase(literalVariables.begin());
818 
819  // Create the undirected edge
820  for (const std::string& var : literalVariables) {
821  variableGraph.insert(firstVariable, var);
822  variableGraph.insert(var, firstVariable);
823  }
824  }
825  }
826 
827  // Find the connected components of G
828  std::set<std::string> seenNodes;
829 
830  // Find the connected component associated with the head
831  std::set<std::string> headComponent;
832  visitDepthFirst(
833  *clause.getHead(), [&](const AstVariable& var) { headComponent.insert(var.getName()); });
834 
835  if (!headComponent.empty()) {
836  variableGraph.visitDepthFirst(*headComponent.begin(), [&](const std::string& var) {
837  headComponent.insert(var);
838  seenNodes.insert(var);
839  });
840  }
841 
842  // Compute all other connected components in the graph G
843  std::set<std::set<std::string>> connectedComponents;
844 
845  for (std::string var : ruleVariables) {
846  if (seenNodes.find(var) != seenNodes.end()) {
847  // Node has already been added to a connected component
848  continue;
849  }
850 
851  // Construct the connected component
852  std::set<std::string> component;
853  variableGraph.visitDepthFirst(var, [&](const std::string& child) {
854  component.insert(child);
855  seenNodes.insert(child);
856  });
857  connectedComponents.insert(component);
858  }
859 
860  if (connectedComponents.empty()) {
861  // No separate connected components, so no point partitioning
862  return;
863  }
864 
865  // Need to extract some disconnected lits!
866  changed = true;
867  std::vector<AstAtom*> replacementAtoms;
868 
869  // Construct the new rules
870  for (const std::set<std::string>& component : connectedComponents) {
871  // Come up with a unique new name for the relation
872  static int disconnectedCount = 0;
873  std::stringstream nextName;
874  nextName << "+disconnected" << disconnectedCount;
875  AstQualifiedName newRelationName = nextName.str();
876  disconnectedCount++;
877 
878  // Create the extracted relation and clause for the component
879  // newrelX() <- disconnectedLiterals(x).
880  auto newRelation = std::make_unique<AstRelation>();
881  newRelation->setQualifiedName(newRelationName);
882  program.addRelation(std::move(newRelation));
883 
884  auto* disconnectedClause = new AstClause();
885  disconnectedClause->setSrcLoc(clause.getSrcLoc());
886  disconnectedClause->setHead(std::make_unique<AstAtom>(newRelationName));
887 
888  // Find the body literals for this connected component
889  std::vector<AstLiteral*> associatedLiterals;
890  for (AstLiteral* bodyLiteral : clause.getBodyLiterals()) {
891  bool associated = false;
892  visitDepthFirst(*bodyLiteral, [&](const AstVariable& var) {
893  if (component.find(var.getName()) != component.end()) {
894  associated = true;
895  }
896  });
897  if (associated) {
898  disconnectedClause->addToBody(souffle::clone(bodyLiteral));
899  }
900  }
901 
902  // Create the atom to replace all these literals
903  replacementAtoms.push_back(new AstAtom(newRelationName));
904 
905  // Add the clause to the program
906  clausesToAdd.push_back(disconnectedClause);
907  }
908 
909  // Create the replacement clause
910  // a(x) <- b(x), c(y), d(z). --> a(x) <- newrel0(), newrel1(), b(x).
911  auto* replacementClause = new AstClause();
912  replacementClause->setSrcLoc(clause.getSrcLoc());
913  replacementClause->setHead(souffle::clone(clause.getHead()));
914 
915  // Add the new propositions to the clause first
916  for (AstAtom* newAtom : replacementAtoms) {
917  replacementClause->addToBody(std::unique_ptr<AstLiteral>(newAtom));
918  }
919 
920  // Add the remaining body literals to the clause
921  for (AstLiteral* bodyLiteral : clause.getBodyLiterals()) {
922  bool associated = false;
923  bool hasVariables = false;
924  visitDepthFirst(*bodyLiteral, [&](const AstVariable& var) {
925  hasVariables = true;
926  if (headComponent.find(var.getName()) != headComponent.end()) {
927  associated = true;
928  }
929  });
930  if (associated || !hasVariables) {
931  replacementClause->addToBody(souffle::clone(bodyLiteral));
932  }
933  }
934 
935  // Replace the old clause with the new one
936  clausesToRemove.push_back(&clause);
937  clausesToAdd.push_back(replacementClause);
938  });
939 
940  // Adjust the program
941  for (AstClause* newClause : clausesToAdd) {
942  program.addClause(std::unique_ptr<AstClause>(newClause));
943  }
944 
945  for (const AstClause* oldClause : clausesToRemove) {
946  program.removeClause(oldClause);
947  }
948 
949  return changed;
950 }

955  // Checks whether an atom is of the form a(_,_,...,_)
956  auto isExistentialAtom = [&](const AstAtom& atom) {
957  for (AstArgument* arg : atom.getArguments()) {
958  if (dynamic_cast<AstUnnamedVariable*>(arg) == nullptr) {
959  return false;
960  }
961  }
962  return true;
963  };

965  // Construct a dependency graph G where:
966  // - Each relation is a node
967  // - An edge (a,b) exists iff a uses b "non-existentially" in one of its *recursive* clauses
968  // This way, a relation can be transformed into an existential form
969  // if and only if all its predecessors can also be transformed.
970  Graph<AstQualifiedName> relationGraph = Graph<AstQualifiedName>();
971 
972  // Add in the nodes
973  for (AstRelation* relation : program.getRelations()) {
974  relationGraph.insert(relation->getQualifiedName());
975  }
976 
977  // Keep track of all relations that cannot be transformed
978  std::set<AstQualifiedName> minimalIrreducibleRelations;
979 
980  auto* ioType = translationUnit.getAnalysis<IOType>();
981 
982  for (AstRelation* relation : program.getRelations()) {
983  // No I/O relations can be transformed
984  if (ioType->isIO(relation)) {
985  minimalIrreducibleRelations.insert(relation->getQualifiedName());
986  }
987  for (AstClause* clause : getClauses(program, *relation)) {
988  bool recursive = isRecursiveClause(*clause);
989  visitDepthFirst(*clause, [&](const AstAtom& atom) {
990  if (atom.getQualifiedName() == clause->getHead()->getQualifiedName()) {
991  return;
992  }
993 
994  if (!isExistentialAtom(atom)) {
995  if (recursive) {
996  // Clause is recursive, so add an edge to the dependency graph
997  relationGraph.insert(clause->getHead()->getQualifiedName(), atom.getQualifiedName());

1000  // it's out of the picture

1002  }

1012  [&](const AstAtom& atom) { minimalIrreducibleRelations.insert(atom.getQualifiedName()); });
1013  });

1016  // A node is reachable in a DFS from an irreducible node if and only if it is
1017  // also an irreducible node
1018  std::set<AstQualifiedName> irreducibleRelations;
1019  for (AstQualifiedName relationName : minimalIrreducibleRelations) {

1021  relationName, [&](const AstQualifiedName& subRel) { irreducibleRelations.insert(subRel); });
1022  }

1026  for (AstRelation* relation : program.getRelations()) {

1028  irreducibleRelations.find(relation->getQualifiedName()) == irreducibleRelations.end()) {
1029  existentialRelations.insert(relation->getQualifiedName());
1030  }
1031  }
1032 
1033  // Reduce the existential relations
1034  for (AstQualifiedName relationName : existentialRelations) {
1035  AstRelation* originalRelation = getRelation(program, relationName);
1036 
1037  std::stringstream newRelationName;
1038  newRelationName << "+?exists_" << relationName;
1039 
1040  auto newRelation = std::make_unique<AstRelation>();
1041  newRelation->setQualifiedName(newRelationName.str());
1042  newRelation->setSrcLoc(originalRelation->getSrcLoc());
1043 
1044  // EqRel relations require two arguments, so remove it from the qualifier
1045  if (newRelation->getRepresentation() == RelationRepresentation::EQREL) {
1046  newRelation->setRepresentation(RelationRepresentation::DEFAULT);
1047  }
1048 
1049  // Keep all non-recursive clauses
1050  for (AstClause* clause : getClauses(program, *originalRelation)) {
1051  if (!isRecursiveClause(*clause)) {
1052  auto newClause = std::make_unique<AstClause>();
1053 
1054  newClause->setSrcLoc(clause->getSrcLoc());
1055  if (const AstExecutionPlan* plan = clause->getExecutionPlan()) {
1056  newClause->setExecutionPlan(souffle::clone(plan));
1057  }
1058  newClause->setHead(std::make_unique<AstAtom>(newRelationName.str()));
1059  for (AstLiteral* lit : clause->getBodyLiterals()) {
1060  newClause->addToBody(souffle::clone(lit));
1061  }
1062 
1063  program.addClause(std::move(newClause));
1064  }
1065  }
1066 
1067  program.addRelation(std::move(newRelation));
1068  }
