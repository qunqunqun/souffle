39 /**
40  * Subclass of Literal that represents the use of a relation
41  * either in the head or in the body of a Clause, e.g., parent(x,y).
42  * The arguments of the atom can be variables or constants.
43  */
44 class AstAtom : public AstLiteral {
45 public:
46  AstAtom(AstQualifiedName name = {}, VecOwn<AstArgument> args = {}, SrcLocation loc = {})
47  : AstLiteral(std::move(loc)), name(std::move(name)), arguments(std::move(args)) {}
48 
49  /** get qualified name */
50  const AstQualifiedName& getQualifiedName() const {
51  return name;
52  }
53 
54  /** get arity of the atom */
55  size_t getArity() const {
56  return arguments.size();
57  }
58 
59  /** set qualified name */
60  void setQualifiedName(AstQualifiedName n) {
61  name = std::move(n);
62  }
63 
64  /** add argument to the atom */
65  void addArgument(Own<AstArgument> arg) {
66  arguments.push_back(std::move(arg));
67  }
68 
69  /** get arguments */
70  std::vector<AstArgument*> getArguments() const {
71  return toPtrVector(arguments);
72  }
73 
74  AstAtom* clone() const override {
75  return new AstAtom(name, souffle::clone(arguments), getSrcLoc());
76  }
77 
78  void apply(const AstNodeMapper& map) override {
79  for (auto& arg : arguments) {
80  arg = map(std::move(arg));
81  }
82  }
83 
84  std::vector<const AstNode*> getChildNodes() const override {
85  std::vector<const AstNode*> res;
86  for (auto& cur : arguments) {
87  res.push_back(cur.get());
88  }
89  return res;
90  }
91 
92 protected:
93  void print(std::ostream& os) const override {
94  os << getQualifiedName() << "(" << join(arguments) << ")";
95  }
96 
97  bool equal(const AstNode& node) const override {
98  const auto& other = static_cast<const AstAtom&>(node);
99  return name == other.name && equal_targets(arguments, other.arguments);
100  }
101 
102  /** name */
103  AstQualifiedName name;
104 
105  /** arguments */
106  VecOwn<AstArgument> arguments;
107 };

235  operation = op;
236  }
237 
238  AstBinaryConstraint* clone() const override {
239  return new AstBinaryConstraint(operation, souffle::clone(lhs), souffle::clone(rhs), getSrcLoc());
240  }
241 
242  void apply(const AstNodeMapper& map) override {
243  lhs = map(std::move(lhs));
244  rhs = map(std::move(rhs));
245  }
246 
247  std::vector<const AstNode*> getChildNodes() const override {
248  return {lhs.get(), rhs.get()};
249  }
250

254  }

258  const auto& other = static_cast<const AstBinaryConstraint&>(node);
259  return operation == other.operation && equal_ptr(lhs, other.lhs) && equal_ptr(rhs, other.rhs);
260  }
