162  /** Get the number of SCCs in the graph. */
163  size_t getNumberOfSCCs() const {

165  }
166 
167  /** Get the SCC of the given relation. */
168  size_t getSCC(const AstRelation* rel) const {

170  }
171 
172  /** Get all successor SCCs of a given SCC. */
173  const std::set<size_t>& getSuccessorSCCs(const size_t scc) const {
174  return successors.at(scc);
175  }
176 
177  /** Get all predecessor SCCs of a given SCC. */
178  const std::set<size_t>& getPredecessorSCCs(const size_t scc) const {
179  return predecessors.at(scc);
180  }
181 
182  /** Get all SCCs containing a successor of a given relation. */
183  std::set<size_t> getSuccessorSCCs(const AstRelation* relation) const {
184  std::set<size_t> successorSccs;
185  const auto scc = relationToScc.at(relation);
186  for (const auto& successor : precedenceGraph->graph().successors(relation)) {
187  const auto successorScc = relationToScc.at(successor);
188  if (successorScc != scc) {
189  successorSccs.insert(successorScc);
190  }
191  }
192  return successorSccs;
193  }
194 
195  /** Get all SCCs containing a predecessor of a given relation. */
196  std::set<size_t> getPredecessorSCCs(const AstRelation* relation) const {
197  std::set<size_t> predecessorSccs;
198  const auto scc = relationToScc.at(relation);
199  for (const auto& predecessor : precedenceGraph->graph().predecessors(relation)) {
200  const auto predecessorScc = relationToScc.at(predecessor);
201  if (predecessorScc != scc) {
202  predecessorSccs.insert(predecessorScc);
203  }
204  }
205  return predecessorSccs;
206  }
207 
208  /** Get all internal relations of a given SCC. */
209  const std::set<const AstRelation*>& getInternalRelations(const size_t scc) const {
210  return sccToRelation.at(scc);
211  }
212 
213  /** Get all external output predecessor relations of a given SCC. */
214  std::set<const AstRelation*> getExternalOutputPredecessorRelations(const size_t scc) const {
215  std::set<const AstRelation*> externOutPreds;
216  for (const auto& relation : getInternalRelations(scc)) {
217  for (const auto& predecessor : precedenceGraph->graph().predecessors(relation)) {
218  if (relationToScc.at(predecessor) != scc && ioType->isOutput(predecessor)) {
219  externOutPreds.insert(predecessor);
220  }
221  }
222  }
223  return externOutPreds;
224  }
225 
226  /** Get all external non-output predecessor relations of a given SCC. */
227  std::set<const AstRelation*> getExternalNonOutputPredecessorRelations(const size_t scc) const {
228  std::set<const AstRelation*> externNonOutPreds;
229  for (const auto& relation : getInternalRelations(scc)) {
230  for (const auto& predecessor : precedenceGraph->graph().predecessors(relation)) {
231  if (relationToScc.at(predecessor) != scc && !ioType->isOutput(predecessor)) {
232  externNonOutPreds.insert(predecessor);
233  }
234  }
235  }
236  return externNonOutPreds;
237  }
238 
239  /** Get all external predecessor relations of a given SCC. */
240  std::set<const AstRelation*> getExternalPredecessorRelations(const size_t scc) const {
241  std::set<const AstRelation*> externPreds;
242  for (const auto& relation : getInternalRelations(scc)) {
243  for (const auto& predecessor : precedenceGraph->graph().predecessors(relation)) {
244  if (relationToScc.at(predecessor) != scc) {
245  externPreds.insert(predecessor);
246  }
247  }
248  }
249  return externPreds;
250  }
251 
252  /** Get all internal output relations of a given SCC. */
253  std::set<const AstRelation*> getInternalOutputRelations(const size_t scc) const {
254  std::set<const AstRelation*> internOuts;
255  for (const auto& relation : getInternalRelations(scc)) {
256  if (ioType->isOutput(relation)) {
257  internOuts.insert(relation);
258  }
259  }
260  return internOuts;
261  }
262 
263  /** Get all internal relations of a given SCC with external successors. */
264  std::set<const AstRelation*> getInternalRelationsWithExternalSuccessors(const size_t scc) const {
265  std::set<const AstRelation*> internsWithExternSuccs;
266  for (const auto& relation : getInternalRelations(scc)) {
267  for (const auto& successor : precedenceGraph->graph().successors(relation)) {
268  if (relationToScc.at(successor) != scc) {
269  internsWithExternSuccs.insert(relation);
270  break;
271  }
272  }
273  }
274  return internsWithExternSuccs;
275  }
276 
277  /** Get all internal non-output relations of a given SCC with external successors. */
278  std::set<const AstRelation*> getInternalNonOutputRelationsWithExternalSuccessors(const size_t scc) const {
279  std::set<const AstRelation*> internNonOutsWithExternSuccs;
280  for (const auto& relation : getInternalRelations(scc)) {
281  if (!ioType->isOutput(relation)) {
282  for (const auto& successor : precedenceGraph->graph().successors(relation)) {
283  if (relationToScc.at(successor) != scc) {
284  internNonOutsWithExternSuccs.insert(relation);
285  break;
286  }
287  }
288  }
289  }
290  return internNonOutsWithExternSuccs;
291  }
292 
293  /** Get all internal input relations of a given SCC. */
294  std::set<const AstRelation*> getInternalInputRelations(const size_t scc) const {
295  std::set<const AstRelation*> internIns;
296  for (const auto& relation : getInternalRelations(scc)) {
297  if (ioType->isInput(relation)) {
298  internIns.insert(relation);
299  }
300  }
301  return internIns;
302  }
303 
304  /** Return if the given SCC is recursive. */
305  bool isRecursive(const size_t scc) const {
306  const std::set<const AstRelation*>& sccRelations = sccToRelation.at(scc);
307  if (sccRelations.size() == 1) {
308  const AstRelation* singleRelation = *sccRelations.begin();
309  if (precedenceGraph->graph().predecessors(singleRelation).count(singleRelation) == 0u) {
310  return false;
311  }
312  }
313  return true;
314  }
315 
316  /** Print the SCC graph. */
317  void print(std::ostream& os) const override;
318 
319 private:
320  PrecedenceGraph* precedenceGraph = nullptr;
321 
322  /** Map from node number to SCC number */
323  std::map<const AstRelation*, size_t> relationToScc;
324 
325  /** Adjacency lists for the SCC graph */
326  std::vector<std::set<size_t>> successors;
327 
328  /** Predecessor set for the SCC graph */
329  std::vector<std::set<size_t>> predecessors;
330 
331  /** Relations contained in a SCC */
332  std::vector<std::set<const AstRelation*>> sccToRelation;
333 
334  /** Recursive scR method for computing SCC */
335  void scR(const AstRelation* relation, std::map<const AstRelation*, size_t>& preOrder, size_t& counter,
336  std::stack<const AstRelation*>& S, std::stack<const AstRelation*>& P, size_t& numSCCs);
337 
338  IOType* ioType = nullptr;
339 };
