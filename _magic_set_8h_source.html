102  friend std::ostream& operator<<(std::ostream& out, const AdornedClause& arg) {
103  size_t currpos = 0;
104  bool firstAdded = true;
105  out << arg.clause->getHead()->getQualifiedName() << "{" << arg.headAdornment << "} :- ";
106 
107  std::vector<AstLiteral*> bodyLiterals = arg.clause->getBodyLiterals();
108  for (AstLiteral* literal : bodyLiterals) {
109  if (auto* atom = dynamic_cast<AstAtom*>(literal)) {
110  if (!firstAdded) {
111  out << ", ";
112  }
113  firstAdded = false;
114  out << atom->getQualifiedName() << "{_}";
115  } else if (auto* neg = dynamic_cast<AstNegation*>(literal)) {
116  if (!firstAdded) {
117  out << ", ";
118  }
119  firstAdded = false;
120  if (currpos < arg.bodyAdornment.size()) {
121  out << neg->getAtom()->getQualifiedName() << "{" << arg.bodyAdornment[currpos++] << "}";
122  } else {
123  out << neg->getAtom()->getQualifiedName() << "{__}";
124  ++currpos;
125  }
126  }
127  }
128  out << ". [order: " << arg.ordering << "]";
129 
130  return out;
131  }

149  void addBinding(const std::string& newVariableName, const AstArgument* arg) {
150  originalArguments[newVariableName] = souffle::clone(arg);
