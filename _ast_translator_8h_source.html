97  struct Location {
98  int identifier{};
99  int element{};
100  std::unique_ptr<RamRelationReference> relation{nullptr};
101 
102  Location() = default;
103 
104  Location(int ident, int elem, std::unique_ptr<RamRelationReference> rel = nullptr)
105  : identifier(ident), element(elem), relation(std::move(rel)) {}
106 
107  Location(const Location& l) : identifier(l.identifier), element(l.element) {
108  if (l.relation != nullptr) {
109  relation = souffle::clone(l.relation);
110  }
111  }
112 
113  Location& operator=(Location other) {
114  identifier = other.identifier;
115  element = other.element;
116  relation = std::move(other.relation);
117  return *this;
118  }
119 
120  bool operator==(const Location& loc) const {
121  return identifier == loc.identifier && element == loc.element;
122  }
123 
124  bool operator!=(const Location& loc) const {
125  return !(*this == loc);
126  }
127 
128  bool operator<(const Location& loc) const {
129  return identifier < loc.identifier || (identifier == loc.identifier && element < loc.element);
130  }
131 
132  void print(std::ostream& out) const {
133  out << "(" << identifier << "," << element << ")";
134  }
135 
136  friend std::ostream& operator<<(std::ostream& out, const Location& loc) {
137  loc.print(out);
138  return out;
139  }
140  };

142  /**
143  * A class indexing the location of variables and record
144  * references within a loop nest resulting from the conversion

147  class ValueIndex {
148  /**
149  * The type mapping variables (referenced by their names) to the
150  * locations where they are used.
151  */
152  using variable_reference_map = std::map<std::string, std::set<Location>>;
153 
154  /**
155  * The type mapping record init expressions to their definition points,
156  * hence the point where they get grounded/bound.
157  */
158  using record_definition_map = std::map<const AstRecordInit*, Location>;
159 
160  /**
161  * A map from generative `AstArgument`s to storage locations. Note,
162  * since in this case AstArgument are indexed by their values (not their
163  * address) no standard map can be utilized.
164  * (By-value indexing induces an ad-hoc form of CSE.)
165  */
166  using generator_location_map = std::vector<std::pair<const AstArgument*, Location>>;
167 
168  /** The index of variable accesses */
169  variable_reference_map var_references;
170 
171  /** The index of record definition points */
172  record_definition_map record_definitions;
173 
174  /** The level of a nested ram operation that is handling a generator operation */
175  generator_location_map arg_generator_locations;
176 
177  public:
178  // -- variables --
179 
180  void addVarReference(const AstVariable& var, const Location& l) {
181  std::set<Location>& locs = var_references[var.getName()];
182  locs.insert(l);
183  }
184 
185  void addVarReference(const AstVariable& var, int ident, int pos,
186  std::unique_ptr<RamRelationReference> rel = nullptr) {
187  addVarReference(var, Location({ident, pos, std::move(rel)}));
188  }
189 
190  bool isDefined(const AstVariable& var) const {
191  return var_references.find(var.getName()) != var_references.end();
192  }
193 
194  const Location& getDefinitionPoint(const AstVariable& var) const {
195  auto pos = var_references.find(var.getName());
196  assert(pos != var_references.end() && "Undefined variable referenced!");
197  return *pos->second.begin();
198  }
199 
200  const variable_reference_map& getVariableReferences() const {
201  return var_references;
202  }
203 
204  // -- records --
205 
206  // - definition -
207 
208  void setRecordDefinition(const AstRecordInit& init, const Location& l) {
209  record_definitions[&init] = l;
210  }
211 
212  void setRecordDefinition(const AstRecordInit& init, int ident, int pos,
213  std::unique_ptr<RamRelationReference> rel = nullptr) {
214  setRecordDefinition(init, Location({ident, pos, std::move(rel)}));
215  }
216 
217  const Location& getDefinitionPoint(const AstRecordInit& init) const {
218  auto pos = record_definitions.find(&init);
219  if (pos != record_definitions.end()) {
220  return pos->second;
221  }
222 
223  fatal("requested location for undefined record!");
224  }
225 
226  // -- generators (aggregates & some functors) --
227 
228  void setGeneratorLoc(const AstArgument& agg, const Location& loc) {
229  arg_generator_locations.push_back(std::make_pair(&agg, loc));
230  }
231 
232  const Location& getGeneratorLoc(const AstArgument& arg) const {
233  // search list
234  for (const auto& cur : arg_generator_locations) {
235  if (*cur.first == arg) {
236  return cur.second;
237  }
238  }
239 
240  fatal("arg `%s` has no generator location", arg);
241  }
242 
243  // -- others --
244 
245  bool isGenerator(const int level) const {
246  // check for aggregator definitions
247  return any_of(arg_generator_locations,
248  [&level](const auto& location) { return location.second.identifier == level; });
249  }
250 
251  bool isSomethingDefinedOn(int level) const {
252  // check for variable definitions
253  for (const auto& cur : var_references) {
254  if (cur.second.begin()->identifier == level) {
255  return true;
256  }
257  }
258  // check for record definitions
259  for (const auto& cur : record_definitions) {
260  if (cur.second.identifier == level) {
261  return true;
262  }
263  }
264  // nothing defined on this level
265  return false;
266  }
267 
268  void print(std::ostream& out) const {
269  out << "Variables:\n\t";
270  out << join(var_references, "\n\t");
271  }
272 
273  friend std::ostream& operator<<(std::ostream& out, const ValueIndex& index) __attribute__((unused)) {
274  index.print(out);

395  /**
396  * Get ram representation of constant.
397  */
398  RamDomain getConstantRamRepresentation(const AstConstant& constant) {
399  if (auto strConstant = dynamic_cast<const AstStringConstant*>(&constant)) {
400  return getSymbolTable().lookup(strConstant->getConstant());
401  } else if (dynamic_cast<const AstNilConstant*>(&constant) != nullptr) {
402  return 0;
403  } else if (auto* numConstant = dynamic_cast<const AstNumericConstant*>(&constant)) {
404  assert(numConstant->getType().has_value());
405  switch (*numConstant->getType()) {
406  case AstNumericConstant::Type::Int:
407  return RamSignedFromString(numConstant->getConstant(), nullptr, 0);
408  case AstNumericConstant::Type::Uint:
409  return RamUnsignedFromString(numConstant->getConstant(), nullptr, 0);
410  case AstNumericConstant::Type::Float: return RamFloatFromString(numConstant->getConstant());
411  }
412  }
413 
414  fatal("unaccounted-for constant");
415  }
